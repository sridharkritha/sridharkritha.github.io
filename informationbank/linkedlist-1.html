// Linked list implementation skeleton
#include <stdlib.h>
#include <string.h>
#include "linkedlist.h"

typedef struct LINKED_LIST_S {
        int data;
        struct LINKED_LIST_S *next;
}LINKED_LIST_S;
LINKED_LIST_S *head, *tail;

// Function to create a linked list and return an opaque handle to it
LINKED_LIST_T linkedListCreate()
{
        // Allocate heap memory for the node
        LINKED_LIST_S *node = (LINKED_LIST_S*)malloc(sizeof(*node)); // *node = for Size of the structure content
        if(node!=NULL)
        {
                // Clear the allocated memory
                memset(node,0,sizeof(*node));
                return node;
        }

        return NULL;
}


// Function to destroy a linked list
// If fn is not NULL then it will call fn for each item in the list
void linkedListDestroy(LINKED_LIST_T list, DestroyFunction fn)
{
        if((list != NULL)&& (fn !=NULL))
        {
                // Destroy the node using the user destroy function
                for(LINKED_LIST_T node=head; node!=NULL; node=node->next)
                        fn(node);
        }
}

// Function to check if a list is empty
// returns 0 - False, there are items in the list
// !0 - True
int linkedListIsEmpty(LINKED_LIST_T list)
{
        // If head and tail are same then list is empty
        if(head != tail)
                return 0;
        return 1;
}

// Append an item to the list. This will add the item to the end of the list
// Item must be non-null
//
// Returns: 0 - Success
// !0 - Failed
int linkedListAppend(LINKED_LIST_T list, void *item)
{
        //list = linkedListCreate();

        if(list!= NULL)
        {
                if(head == NULL)
                { // If it is the first node
                        list->data = (int)item;
                        list->next = NULL;
                        head = tail = list;
                        return 0;
                }
                LINKED_LIST_T temp = NULL;
                temp = linkedListCreate();
                temp->data = (int)item;
                temp->next = NULL;


                // if it is not a first node then consider as tail node
                // and change the tail pointer
                tail->next = temp;
                tail = tail->next; // Move to tail pointer
                return 0;
        }

        return -1;

}

// Insert an item into the list. This will insert the item in front of the
// Current item in the list. Item must be non-null
// Returns: 0 - Success
// !0 - Failed
int linkedListInsert(LINKED_LIST_T list, void *item)
{
        if(list!= NULL)
        {
                list->data = (int)item;
                list->next = NULL;

                if(head == NULL)
                {
                        // if it is a first node then both head/tail points to it
                        head = tail = list;
                        return 0;
                }
                // if it is not a first node then make this node as a head node
                list->next = head;
                head = list; // Move the head ptr to first loc

                return 0;
        }

        return -1;
}

// Remove the current item from the list
// Returns:
// NULL - the list is empty
// item - The removed item
void * linkedListRemove(LINKED_LIST_T list)
{
        int revodeData =0;
        if(list!=NULL)
        {
                revodeData = list->data;

                LINKED_LIST_T prevNode = head;
                LINKED_LIST_T temp = head;
                // Find the prev node of the list to be removed
                for(temp=head; temp!=list; temp=temp->next)
                {
                        prevNode = temp;
                }
                // get the next node addr before deleting the Node
                // for assigning into prevNode next
                prevNode->next = temp->next ;
                // Reset the tail pointer to head pos before deleting the node
                if(prevNode== head)
                        tail = head;
                free(temp);
                list = NULL;
                return (void*)revodeData;

        }
        return NULL;
}

// Get the current item in the list
// Returns:
// NULL - The list is empty
// item - The current item
void * linkedListCurrent(LINKED_LIST_T list)
{
        // Return the data if the list is not NULL
        if(list!=NULL)
                return (void*)list->data;
        return NULL;
}

// Get the first/last item from the list and set as current
//
void * linkedListGetFirst(LINKED_LIST_T list)
{
        if(list!=NULL)
        {
                // Retrun the first ele (head) if the list is not empty
                return (void*)head->data;
        }

        return NULL;

}

void * linkedListGetLast(LINKED_LIST_T list)
{
        if(list!=NULL)
        {
                // Retrun the last ele (tail) if the list is not empty
                return (void*)tail->data;
        }
        return NULL;
}

// Get the next / previous item from the list and set as
// current
void * linkedListGetNext(LINKED_LIST_T list)
{
        //int nextEle =0;
        if(list!=NULL)
        {
                // Move to next node and return the value if it is not NULL
                //nextEle = list->data;
                list = list->next;
                if(list!=NULL)
                        return (void*)list->data;
        }
        return NULL;
}

void * linkedListGetPrev(LINKED_LIST_T list)
{
        int prevData=0;
        if(list!=NULL)
        {
                // Find the prev node of the given node and return the data
                for( LINKED_LIST_T temp =head; temp!=list; temp=temp->next )
                        prevData = temp->data;

                return (void*)prevData;
        }
        return NULL;
}

// Sort the linked list
// fn is a function used to compare two list items
// usr_info is a value passed to every call of fn
int linkedListSort(LINKED_LIST_T list, CompareFunction fn, void *usr_info)
{
        int lenList =0;
        int x =0;
        // Finding the length of the list
        for(LINKED_LIST_T temp=head; temp!=NULL; temp=temp->next)
        {
                lenList = lenList+1;
        }
        // Sorting the linked list w.r.t to comp.fun fn and usr_info
        for (int p = 0; p < lenList - 1; p++) {
                for(int i = 0; i < lenList - 1; i++)
                {
                        LINKED_LIST_T nextEle= list->next;
                        if(nextEle !=NULL)
                        {
                                // Call the user defined compare function
                                if(fn((void*)list->data,(void*)nextEle->data,usr_info))
                                {
                                        // Swap the data
                                        x = list->data;
                                        list->data = nextEle->data;
                                        nextEle->data = x;
                                }
                        }
                }
        }

        return 1;
}
