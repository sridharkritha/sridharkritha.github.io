Traversing Matrix in all possible direction:

1. Traverse in row by row [Top to Bottom][Left to Right] 


for(int rr = 0; rr < r; ++rr)	
  for(int cc = 0; cc < c; ++cc) 
     cout<<hex<< matrix[rr][cc] <<" ";	 
	/*
	1 2 3 4
	5 6 7 8
	9 a b c
	*/

    cout<<hex<< matrix[rr][cc] <<" ";

int main()
{
	/*
	int matrix[3][4] = { 1, 2, 3, 4,
	5, 6, 7, 8,
	9, a, b, c };
	*/
	int r = 3,c = 4,i=0;
	vector<vector<int> > matrix(r,vector<int>(c,0));
	// Assign consecutive values to each cell in row by row
	for(int rr = 0; rr < r; ++rr) for(int cc = 0; cc< c; ++cc) matrix[rr][cc] = ++i;
 
	cout<<"Traverse in row by row [Top to Bottom][Left to Right]"<<endl;
	for(int rr = 0; rr < r; ++rr)
	{
		for(int cc = 0; cc < c; ++cc) cout<<hex<<matrix[rr][cc]<<" ";
		cout<<endl;
	}
 
	/*
	1 2 3 4
	5 6 7 8
	9 a b c
	*/
	cout<<"-----------------------------------"<<endl;
 
	// ROW BY ROW
	cout<<"Traverse by [Bottom to Top][Right to Left][row by row]"<<endl;
	for(int rr = r-1; rr > -1; --rr)
	{
		for(int cc = c-1; cc > -1; --cc) cout<<hex<<matrix[rr][cc]<<" ";
		cout<<endl;
	}
	/*
	c b a 9
	8 7 6 5
	4 3 2 1
	*/
	cout<<"-----------------------------------"<<endl;
 
	cout<<"Traverse by [Top to Bottom][Right to Left][row by row]"<<endl;
	for(int rr = 0; rr < r; ++rr)
	{
		for(int cc = c-1; cc > -1; --cc) cout<<hex<<matrix[rr][cc]<<" ";
		cout<<endl;
	}
	/*
	4 3 2 1
	8 7 6 5
	c b a 9
	*/
	cout<<"-----------------------------------"<<endl;
 
	cout<<"Traverse by [Bottom to Top][Left to Right][row by row]"<<endl;
	for(int rr = r-1; rr > -1; --rr)
	{
		for(int cc = 0; cc < c; ++cc) cout<<hex<<matrix[rr][cc]<<" ";
		cout<<endl;
	}
	/*
	9 a b c
	5 6 7 8
	1 2 3 4
	*/
	cout<<"-----------------------------------"<<endl;
 
	// COLUMN BY COLUMN
	cout<<"Traverse by [Top to Bottom][Left to Right][column by column]"<<endl;
	for(int cc = 0; cc < c; ++cc)
	{
		for(int rr = 0; rr < r; ++rr) cout<<hex<<matrix[rr][cc]<<" ";
		cout<<endl;
	}
	/*
	1 5 9
	2 6 a
	3 7 b
	4 8 c
	*/
	cout<<"-----------------------------------"<<endl;
 
	cout<<"Traverse by [Bottom to Top][Right to Left][column by column]"<<endl;
	for(int cc = c-1; cc > -1; --cc)
	{
		for(int rr = r-1; rr > -1; --rr) cout<<hex<<matrix[rr][cc]<<" ";
		cout<<endl;
	}
	/*
	c 8 4
	b 7 3
	a 6 2
	9 5 1
	*/
	cout<<"-----------------------------------"<<endl;
 
	cout<<"Traverse by [Bottom to Top][Left to Right][column by column]"<<endl;
	for(int cc = c-1; cc > -1; --cc)
	{
		for(int rr = 0; rr < r; ++rr) cout<<hex<<matrix[rr][cc]<<" ";
		cout<<endl;
	}
	/*
	4 8 c
	3 7 b
	2 6 a
	1 5 9
	*/
	cout<<"-----------------------------------"<<endl;
 
	cout<<"Traverse by [Top to Bottom][Right to Left][column by column]"<<endl;
	for(int cc = 0; cc < c; ++cc)
	{
		for(int rr = r-1; rr > -1; --rr) cout<<hex<<matrix[rr][cc]<<" ";
		cout<<endl;
	}
	/*
	9 5 1
	a 6 2
	b 7 3
	c 8 4
	*/
	cout<<"-----------------------------------"<<endl;
 
	// http://stackoverflow.com/questions/1779199/traverse-matrix-in-diagonal-strips
	// http://stackoverflow.com/questions/2112832/traverse-rectangular-matrix-in-diagonal-strips?rq=1
 
	cout<<"Rectangular Matrix - Diagonally top left to bottom right"<<endl;
	for (int slice = 0; slice < r + c - 1; ++slice) {
		printf("Slice %d: ", slice);
		int z1 = slice < c ? 0 : slice - c + 1;
		int z2 = slice < r ? 0 : slice - r + 1;
		for (int j = z1; j <= slice - z2; ++j)
		{
			cout<< matrix[j][slice - j]<<" ";
		}
		cout<<endl;
	}

	Slice 0: 1
	Slice 1: 2 5
	Slice 2: 3 6 9
	Slice 3: 4 7 a
	Slice 4: 8 b
	Slice 5: c

	cout<<"-----------------------------------"<<endl;
 
	cout<<"Diagonally bottom right to top left"<<endl;
	for (int slice = r + c - 2; slice > -1; --slice) // Slice = No.of Diagonal strips
	{
		printf("Slice %d: ", slice);
		int z1 = slice < c ? 0 : slice - c + 1;
		int z2 = slice < r ? 0 : slice - r + 1;
		for (int j = z1; j <= slice - z2; ++j)
		{
			cout<< matrix[j][slice - j]<<" ";
		}
		cout<<endl;
	}
	/*
	Slice 5: c
	Slice 4: 8 b
	Slice 3: 4 7 a
	Slice 2: 3 6 9
	Slice 1: 2 5
	Slice 0: 1
	*/
	cout<<"-----------------------------------"<<endl;
 
	cout<<"Diagonally top right to bottom left"<<endl;
	for (int slice = 0; slice < r + c - 1; ++slice) // Slice = No.of Diagonal strips
	{
		printf("Slice %d: ", slice);
		// Upper and Lower Triangle
		int z1 = slice < c ? 0 : slice - c + 1;
		int z2 = slice < r ? 0 : slice - r + 1;
		for (int j = z1; j <= slice - z2; ++j)
		{
			cout<< matrix[j][(c - 1)-(slice - j)]<<" ";
		}
		cout<<endl;
	}
	/*
	Slice 0: 4
	Slice 1: 3 8
	Slice 2: 2 7 c
	Slice 3: 1 6 b
	Slice 4: 5 a
	Slice 5: 9
	*/
	cout<<"-----------------------------------"<<endl;
 
	cout<<"Diagonally bottom left to top right"<<endl;
	for (int slice = r + c - 2; slice > -1; --slice) // Slice = No.of Diagonal strips
	{
		printf("Slice %d: ", slice);
		// Upper and Lower Triangle
		int z1 = slice < c ? 0 : slice - c + 1;
		int z2 = slice < r ? 0 : slice - r + 1;
		for (int j = z1; j <= slice - z2; ++j)
		{
			cout<< matrix[j][(c - 1)-(slice - j)]<<" ";
		}
		cout<<endl;
	}
	/*
	Slice 5: 9
	Slice 4: 5 a
	Slice 3: 1 6 b
	Slice 2: 2 7 c
	Slice 1: 3 8
	Slice 0: 4
	*/
	cout<<"-----------------------------------"<<endl;
 
	// Grid point (r+1) x (c+1)
	return 0;
}
