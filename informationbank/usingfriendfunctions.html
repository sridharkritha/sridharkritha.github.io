Overloading operators using Friend functions
 
 

C doesn't support any form of Overloading. Ex: Functional Overloading is not supported in C but supported in C++ using Name Mangling(Decoration).   [Thanks to Sivabalan who reminds me !]


Using Friend function:            

    When the operator does not modify its operands, the best way to overload the operator is via friend function. 


Example: 

    class Cents {
        private:
            int m_nCents; 
        public:
            Cents(int nCents) { m_nCents = nCents; }            
            friend Cents operator+(Cents &c1, Cents &c2); // Add Cents + Cents
            int GetCents() { return m_nCents; }
    };

    // note: this function is not a member function! 
    Cents operator+(Cents &c1, Cents &c2) {        
        return Cents(c1.m_nCents + c2.m_nCents); // use the Cents constructor and operator+(int, int)
    }

    int main() {
        Cents cCents1(6);
        Cents cCents2(8);
        Cents cCentsSum = cCents1 + cCents2;
        std::cout << "I have " << cCentsSum .GetCents() << " cents." << std::endl;
    }

Overloading operators for operands of different types(User-defined & Primitive):

For example, if we have user-defined datatype Cents(4), we may want to add the primitive data type integer 6 to this to produce the result Cents(10). 

Note: If operands are different types, then x + y is not the same as y + x. For example, Cents(4) + 6 would call operator+(Cents, int), and 6 + Cents(4) would call operator+(int, Cents). Consequently, whenever we overload binary operators for operands of different types, we actually need to write two functions one for each case.  


Here is an example of that: 

class Cents
{
private:
    int m_nCents;
public:
    Cents(int nCents) { m_nCents = nCents; }
    friend Cents operator+(Cents &cCents, int nCents);  // Overload cCents + int 
    friend Cents operator+(int nCents, Cents &cCents);  // Overload int + cCents  
    friend Cents operator+(Cents &cCents,Cents &dCents) // Overload cCents + dCents 
    int GetCents() { return m_nCents; }
};

// Note: this function is not a member function! 
Cents operator+(Cents &cCents, int nCents) {
    return Cents(cCents.m_nCents + nCents);
}

/* Method I */ 
// Note: this function is not a member function!
Cents operator+(int nCents, Cents &cCents) {
    return Cents(cCents.m_nCents + nCents);
}

/* Method II */
Cents operator+(int nCents, Cents &cCents) {    
    return (cCents + nCents);    // Call-> Cents operator+(Cents &cCents, int nCents)
}

// Note: this function is not a member function!
Cents operator+(Cents &cCents,Cents &dCents) {
    return Cents(cCents.m_nCents + dCents.m_nCents);
}

int main() {

    Cents c1 = Cents(4) + 6;  // Call operator+(Cents &cCents, int nCents)
    Cents c2 = 6 + Cents(4); // operator+(int nCents, Cents &cCents) 
    Cents c3 = 6 + Cents(4)+3+Cents(8); // operator+(Cents &cCents,Cents &dCents)
    std::cout << "I have " << c1.GetCents() << " cents." << std::endl;
    std::cout << "I have " << c2.GetCents() << " cents." << std::endl;
    std::cout << "I have " << c3.GetCents() << " cents." << std::endl;
}

Overloading the I/O operators - Overloading << and >> Operators:

    class Point {
        private:
            double m_dX, m_dY, m_dZ; 
        public:
            Point(double dX=0.0, double dY=0.0, double dZ=0.0) {
                m_dX = dX;
                m_dY = dY;
                m_dZ = dZ;
            } 

            friend ostream& operator<< (ostream &out, Point &cPoint);
            friend istream& operator>> (istream &in, Point &cPoint);

            double GetX() { return m_dX; }
            double GetY() { return m_dY; }
            double GetZ() { return m_dZ; }
    };

    ostream& operator<< (ostream &out, Point &cPoint) {
        out << cPoint.GetX();
        out << cPoint.GetY();
        out << cPoint.GetZ();
        return out;
    }

    istream& operator>> (istream &in, Point &cPoint) {
        in >> cPoint.m_dX;
        in >> cPoint.m_dY;
        in >> cPoint.m_dZ;
        return in;
    } 

    int main() {
        cout << "Enter a point: " << endl; 
        Point cPoint;
        cin >> cPoint; 
        cout << "You entered: " << cPoint << endl; 
    }

Explanation: 

     Consider the expression cout << cPoint. If the operator is <<, what are the operands? The left operand is the cout object, and the right operand is your Point class object. cout is actually an object of type ostream. Therefore, our overloaded function will look like this: 

     friend ostream& operator<< (ostream &out, Point &cPoint);

The only tricky part here is the return type. Why are we returning an object of type ostream? The answer is that we do this so we can “chain” output commands together, such as cout << cPoint << endl;

     Consider what would happen if our operator<< returned void. When the compiler evaluates cout << cPoint << endl; due to the precedence/associativity rules, it evaluates this expression as (cout << cPoint) << endl; cout << cPoint calls our void-returning overloaded operator<< function, which returns void. Then the partially evaluated expression becomes: void << endl; which makes no sense!

 

Note:  The overloaded output operator<<

     friend ostream& operator<< (ostream &out, Point &cPoint);

             is actually better written as

     friend ostream& operator<<(ostream &out, const Point &cPoint); 

 

However, for the overloaded input operator>>, you will have to leave cPoint as non-const because the overloaded operator>> modifies cPoints members.


Overloading unary operators:

The ! Operator is often used as a shorthand method to test if something is set to a non-zero value. 

For example, the following example would only execute if nX is non-zero: 

            if (nX) {

                // do something 

             } 

              

Similarly, we can overload the! Operator to work similarly for a user-defined class.

    class Point {
    private:
        double m_dX, m_dY, m_dZ;
    public:
        Point(double dX=0.0, double dY=0.0, double dZ=0.0) {
            m_dX = dX;
            m_dY = dY;
            m_dZ = dZ;
        }

        friend Point operator- (const Point &cPoint); // Convert a Point into it's negative equivalent    
        friend bool operator! (const Point &cPoint);  // Return true if the point is set at the origin
    };

    // Convert a Point into it's negative equivalent
    Point operator- (const Point &cPoint) {
        return Point(-cPoint.m_dX, -cPoint.m_dY, -cPoint.m_dZ);
    }

    // Return true if the point is set at the origin
    bool operator! (const Point &cPoint) {
        return (cPoint.m_dX == 0.0 && cPoint.m_dY == 0.0 && cPoint.m_dZ == 0.0);
    }

    int main() {
        Point cPoint; // use default contructor to set to (0.0, 0.0, 0.0) 
        if (!cPoint) cout << "cPoint was set at the origin." << endl;
        else         cout << "cPoint was not set at the origin." << endl; 

        cPoint(5.0,6.1,7.4);
        -cPoint; 

        if (!cPoint) cout << "cPoint was set at the origin." << endl;
        else         cout << "cPoint was not set at the origin." << endl; 
    }
