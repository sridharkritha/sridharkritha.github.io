Loading the Shader file:
Method 1:
Explicit Compilation  and loading binary- using D3DXCreateEffect:
// Method 1: pre-compiled loading
		  FILE * pFile;
		  long lSize;
		  size_t result;
 
		  pFile = fopen ( "Media\\ss.fxo" , "rb" );
		  if (pFile==NULL) {fputs ("File error",stderr); exit (1);}
 
		  // obtain file size:
		  fseek (pFile , 0 , SEEK_END);
		  lSize = ftell (pFile);
		  rewind (pFile);
 
		  // Allocate memory to contain the whole file:
 
		  // Method 1: C - Style - Malloc
		  char * buffer;
		  buffer = (char*) malloc (sizeof(char)*lSize);
		  if (buffer == NULL) {fputs ("Memory error",stderr); exit (2);}
		  // copy the file into the buffer:
		  result = fread (buffer,1,lSize,pFile);
		  if (result != lSize) {fputs ("Reading error",stderr); exit (3);}
		  /* the whole file is now loaded in the memory buffer. */
		  D3DXCreateEffect(pWin32DispDriver->dObjs->g_pd3dDevice,buffer, lSize,NULL, NULL, D3DXFX_NOT_CLONEABLE,NULL, &g_pEffect, &pBuffer );
		  // terminate
		  fclose (pFile);
		  free (buffer);
 
		  // Method 2: C++ Style - Vector
		  std::vector<unsigned char> Cache(lSize);
		  // copy the file into the buffer:
		  result = fread (&(Cache)[0],1,lSize,pFile);
		  if (result != lSize) {fputs ("Reading error",stderr); exit (3);}
		  /* the whole file is now loaded in the memory buffer. */
		  D3DXCreateEffect(pWin32DispDriver->dObjs->g_pd3dDevice,&(Cache)[0], lSize,NULL, NULL, D3DXFX_NOT_CLONEABLE,NULL, &g_pEffect, &pBuffer );
		  // terminate
		  // Vector deallocation will happean atomatically

Data Type:
Vector Type:
------------
	A vector contains between one and four scalar components.
 
You could use any of the following vector declarations: 
 
float4 fVector;
float  fVector[4];
vector fVector;  // by default all 4 components are float type
vector <float, 4> fVector;  
 
To declare a 3-tuple of bools, for example, you could use any of the following declarations:
 
bool3 bVector0;
bool bVector1[3];
vector <bool, 3> bVector2; 
 
Matrix:
-------
float4x4 view_proj_matrix; 
matrix<float, 4, 4> view_proj_matrix; 
	
 
In the swizzle case, the components must come from either the { x,y,z,w } or {r, g, b, a}
 
float4 pos = {3.0f, 5.0f, 2.0f, 1.0f};
float value0 = pos[0]; // value0 is 3.0f
float value1 = pos.x; // value1 is 3.0f
float value2 = pos.g; // value2 is 5.0f
float2 vec0 = pos.xy; // vec0 is {3.0f, 5.0f}
float2 vec1 = pos.ry; // INVALID because of bad swizzle

Texture data in HLSL: 

Usually, we apply the following code to send texture to GPU

     device->SetTexture(0, texture0);
If we only set texture once, then we can simply get this texture data in HLSL by code
   samplerTextureSampler;
To get the color from this sampler, we can use HLSLfunctiontex2D(...)
   float color = tex2D(TextureSampler, textureCoords)
This operation should be called in pixel shader function not in vertex shader function.
If we have multiple Texture objects in application,such as
   Texturet1, t2;
We can send them to GPU by code in application
   effect->SetValue("Texture1", t1);
   effect->SetValue("Texture2", t2);
Then in HLSL, we can get the Textures data back in this way:
   TextureTexture1;
   TextureTexture2;
However, we still have to make samplers from them like
    samplersampler1 = sampler_state
    {
        texture= <Texture1>;
    }

    sampler sampler2 = sampler_state
    {
        texture= <Texture2>;
        minfilter= LINEAR;
        mipfilter= LINEAR;
        magfilter= LINEAR;
    } 

D3DXSprite & Effect Call flow
void Render()
{
	unsigned int passes;
	gD3dDevice->Clear(0, 0, D3DCLEAR_TARGET | D3DCLEAR_ZBUFFER, 0x00000000, 1.0f, 0);
	gD3dDevice->BeginScene();
	gSprite->Begin(0);
	g_pEffect->SetTechnique("PostProcess");
	g_pEffect->SetTexture( "Tex0", gTextureBackdrop );
	float blurFactor = 25;
	g_pEffect->SetValue("TextureBlur",&blurFactor ,sizeof(float));
	g_pEffect->Begin(&passes, 0);
	for(unsigned int pass = 0; pass < passes; ++pass)
	{
		g_pEffect->BeginPass(pass);
		D3DXVECTOR3 spritePos(0.0f, 0.0f, 0.0f);
		gD3dDevice->SetTexture(0,gTextureBackdrop);
		gSprite->Draw(gTextureBackdrop, 0, 0, &spritePos, 0xffffffff);
		gSprite->End();
		g_pEffect->CommitChanges();
		g_pEffect->EndPass();
	}
	g_pEffect->End();
	gD3dDevice->EndScene();
	gD3dDevice->Present(NULL,NULL,NULL,NULL);
}
