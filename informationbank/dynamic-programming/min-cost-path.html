Min Cost Path

Ref:


Given a cost matrix cost[][] and a position (m, n) in cost[][], write a function that returns cost of minimum cost path to reach (m, n) from (0, 0). Each cell of the matrix represents a cost to traverse through that cell. Total cost of a path to reach (m, n) is sum of all the costs on that path (including both source and destination). You can only traverse down, right and diagonally lower cells from a given cell, i.e., from a given cell (i, j), cells (i+1, j), (i, j+1) and (i+1, j+1) can be traversed. You may assume that all costs are positive integers.

For example, in the following figure, what is the minimum cost path to (2, 2)?

The path with minimum cost is highlighted in the following figure. The path is (0, 0) –> (0, 1) –> (1, 2) –> (2, 2). The cost of the path is 8 (1 + 2 + 2 + 3).

1) Recursive Solution and Dynamic Programming Approach:

int min2(int a, int b)
{
	return (a < b)? a : b;
}
 
int min(int a, int b, int c)
{
	return (min2(a,b) < c)? min2(a,b) : c;
}
 
// Recursive Method
int minCost(vector<vector<int> > cost, int r, int c) {
	if (r < 0 || c < 0)
		return INT_MAX; // Some Large value
 
	if((r==0)&&(c==0)) {
		return cost[0][0];
	}
 
	return min(minCost(cost,r-1,c),minCost(cost,r,c-1),minCost(cost,r-1,c-1)) + cost[r][c];
}
 
// Dynamic Programming
int minCostDynPrg(vector<vector<int> > cost, int r, int c)
{
	 int i, j;
 
	 r = r + 1; // Note: Extra Element
	 c = c + 1;
 
	 // Instead of following line, we can use int tc[m+1][n+1] or
	 // dynamically allocate memoery to save space. The following line is
	 // used to keep te program simple and make it working on all compilers.	 
	 vector<vector<int> > tc( r, vector<int>(c, 0) );
 
	 tc[0][0] = cost[0][0];
 
	 /* Initialize first column of total cost(tc) array */
	 for (i = 1; i < r; i++)
		tc[i][0] = tc[i-1][0] + cost[i][0];
 
	 /* Initialize first row of tc array */
	 for (j = 1; j < c; j++)
		tc[0][j] = tc[0][j-1] + cost[0][j];
 
	 /* Construct rest of the tc array */
	 for (i = 1; i < r; i++)
		for (j = 1; j < c; j++)
			tc[i][j] = min(tc[i-1][j-1], tc[i-1][j], tc[i][j-1]) + cost[i][j];
 
	 return tc[r-1][c-1];
}
 
int main() {
 
	int cost[][3] = { {1, 2, 3},
					  {4, 8, 2},
					  {1, 5, 3}
					};
 
	int nRows = sizeof cost / sizeof cost[0];
 
	int nCols = sizeof cost[0] / sizeof(int);
 
	// 2d matrix - n, m known only at run time
	vector<vector<int> > vCost( nRows, vector<int>(nCols, 0) );
 
	// Copy 2d Array to 2d Vector
	for (size_t i = 0; i < nRows; ++i)
		vCost[i].assign(cost[i], cost[i] + nCols);
 
	int minPathCost = minCost(vCost, 2, 2);
	cout<<minPathCost;
 
	minPathCost = minCostDynPrg(vCost, 2, 2);
	cout<<minPathCost;
 
	return 0;
}
