function.arguments(deprecated):
The syntax function.arguments is deprecated. The recommended way to access the arguments object available within functions is simply to refer to the variable arguments.
    function hai(a,b) 
    {
        console.log(hai.arguments); // deprecated - [5, 10, callee: ƒ,] - Array like Object
    }
    hai(5,10);

arguments( use only for NON arrow functions):
arguments is an Array-like object accessible inside functions that contains the values of the arguments passed to that function.
    function hai(a,b) 
    {
        console.log(arguments); // OK - [5, 10, callee: ƒ,] - Array like Object
    }

    hai(5,10);

Arrow functions:
Arrow functions don't have arguments object inside the function. ES6 has solved with a rest parameter:
    const hai = (...args) => console.log ("arguments", args); // [5, 10] - Array (original)
    hai (5,10);

Object Literals: Dynamic key name
    const mystery = "answer";
    const PI = Math.PI;

    const obj = {
        age: 10,
        f1() { },         // function inside an object
        f2: () => { },    // arrow function inside an object
        [mystery]: 43,   // "answer": 43, (Dynamically 'mystery' will evaluated and then it become property)

    };

    console.log(obj.mystery); // undefined
    console.log(obj.answer);  // 43

Destructuring assignment syntax : Extract data from arrays or objects using a syntax that mirrors the construction of array and object literals.

    Object destructuring:
    var o = {p: 42, q: true};
    // Extract
    var {p, q} = o; // p = 42, q = true

    // Assign new variable names
    var {p: foo, q: bar} = o; // foo = 42, bar = true

    // without destructuring
    var StyleSheet = React.StyleSheet;
    var Text       = React.Text;
    // Rewritten as
    var { StyleSheet, Text } = React;

    Array destructuring:
var foo = [1,2,3,4,5];

    // without destructuring
    var one   = foo[0]; // 1
    var two   = foo[1]; // 2
    var three = foo[2]; // 3

    // with destructuring
    var [one, two, three] = foo;

    const [, two, , four] = [1, 2, 3, 4, 5]; // skipping 
    console.log(two, four); // 2,4

    };

    // Extract ONLY "radius" property from the circle object using destructuring syntax {}
    const circleArea = ({ radius }) => (Math.PI * radius * radius).toFixed(2);   // 12.57
    console.log( circleArea(circle) );

    const sphereArea = ({ radius }, {precision}) => (Math.PI * radius * radius).toFixed(precision); // 12.56637
    console.log( sphereArea(circle, {precision: 5}) ); // Passing extra argument

    const sphereAreaDefault_1 = ({ radius }, {precision = 2}) => (Math.PI * radius * radius).toFixed(precision); // 12.57
    console.log( sphereAreaDefault_1(circle, {}) ); // Passing EMPTY extra argument and use default parameters

    const sphereAreaDefault_2 = ({ radius }, {precision = 2} = {}) => (Math.PI * radius * radius).toFixed(precision); // 12.56637
    console.log( sphereAreaDefault_2(circle) ); // Passing NO extra argument and use default parameters

ES6 Class Sugar Vs ES5 Function Comparison:
ES6 keywords to remember    
    1. class
    2. constructor (optional)
    2. extends
    3. super (optional) - can act as a constructor (or) an object
    4. NO 'function' keyword inside the class.

    // ES6
    class Animal {
        constructor(name) { this.name = name; }
        // function eat() { } // ERROR: 'function' keyword is NOT allowed inside class
        eat() { console.log(`${this.name} is eating APPLES!`); }
    }

    // Old way - ES5
    function Animal(name) { this.name = name; }
    Animal.prototype.eat = function() { console.log(`${this.name} is eating APPLES!`); }

    // ES6
    class Gorilla extends Animal {
        constructor(name) { // whole constructor is OPTIONAL bcos of 'extends' 
            super(name);    // 'super' as FUNCTION. Call the parent constructor.
        }
        eat() { console.log(`${this.name} is eating BANANA!`); }
        tellMore() {
            super.eat(); // 'super' as OBJECT. George is eating APPLES!
            this.eat();  // George is eating BANANA!
        }
    }

    // Old way - ES5
    function Gorilla(name) {
        Animal.call(this, name); // 'SUPER' role
    }

    Gorilla.prototype = Object.create(Animal.prototype); // 'EXTENDS' role
    Gorilla.prototype.constructor = Gorilla;

    Gorilla.prototype.eat = function() { console.log(`${this.name} is eating BANANA!`); };
    Gorilla.prototype.tellMore = function () {
        Animal.prototype.eat.call(this);
        this.eat();
    };

    // Usage
    var gorilla = new Gorilla('George'); // automatically calls ALL the constructors from CHILD to PARENT
    gorilla.eat();
    gorilla.tellMore();
