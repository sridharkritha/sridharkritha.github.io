Iterable Objects: String, Array, array-like objects (arguments, NodeList), Map, Set, and user-defined iterables.

Iterable allows us to make any object useable in a for..of loop.
Many built-in objects (Array, String) are iterable but NOT all objects are iterable.

for..of:
    1. for..of automatically do one off call to  'range[Symbol.iterator]()' to get an object with next() method.
    2. On each iteration, the next() method called to get the object of form { done:false/true, value: somevalue }

    let range = { from: 1, to: 5 };
    range[Symbol.iterator] = function () // called once, in the beginning of for..of
    {
        return {
                    current: this.from,
                    last: this.to,
                    next() // called for every iteration by for..of
                    {
                      if (this.current <= this.last) return { done: false, value: this.current++ };
                      else return { done: true };
                    }
        };
    };

    // now it works!
    for (let num of range) {
        alert(num); // 1, 2, 3, 4, 5
    }

Rewrite the previous range object iterator using geneator ( It is easy to write the iterator using generator)
    let range = {
        from: 1,
        to: 5,
        *[Symbol.iterator]() { // a shorthand for [Symbol.iterator]: function*()
            for (let value = this.from; value <= this.to; value++) {
                yield value;
            }
        }
    };

    alert([...range]); // 1,2,3,4,5
That works, because range[Symbol.iterator]() now returns a generator, and generator methods are exactly what for..of expects:
    1. it has a .next() method
    2. that returns object in the form {value: ..., done: true/false}
That’s not a coincidence, of course. Generators were added to JavaScript language with iterators in mind, to implement them easily.

Generators:  
    1. Regular function return only one single value (or nothing).
    2. Generator function can return (“yield”) multiple values, one after another, on-demand. They work great with iterables.
    3. To create a generator, we need a special syntax construct: function*, so-called “generator function”.
    4. Generator functions behave differently from regular ones. When such function is called, it doesn’t run its code. Instead it returns
         a special object, called “generator object”, to manage the execution.
    function* generateSequence() {
        yield 1;
        yield 2;
        return 3; // return NOT yield
    }
    
    let generator = generateSequence(); // "generator function" creates "generator object"
    alert(generator); // [object Generator]
The main method inside the generator object is next(). When called, it runs the execution until the nearest yield <some value or blank> statement). Then the function execution pauses, and the yielded value is returned to the outer code.

The result of next() is always an object with 2 properties:
    value: the yielded value.
    done: true if the function code has finished, otherwise false.
    let one = generator.next();
    alert(JSON.stringify(one)); // {value: 1, done: false}

    let two = generator.next();
    alert(JSON.stringify(two)); // {value: 2, done: false}

    let three = generator.next();
    alert(JSON.stringify(three)); // {value: 3, done: true}

Generators are iterable: So you can loop over their values using for..of:
    for(let value of generator) {
        alert(value); // 1, then 2
    }
    // It doesn’t show 3!

It’s because for..of iteration ignores the last value, when done: true. So, if we want all results to be shown by for..of, we must return them with yield:
    function* generateSequence() {
        yield 1;
        yield 2;
        yield 3; // yield NOT return
    }

    let generator = generateSequence();

    for (let value of generator) {
        alert(value); // 1, then 2, then 3
    }

        //  get() method can intercepts and access the properties in the target.
        get: function (target, prop, receiver) { return "world"; }
    };

    const proxy2 = new Proxy(target, handler2);
    console.log(proxy2.message1); // world
    console.log(proxy2.message2); // world

1. Calling Functions by Proxy

Wrapper using plain function:

    }

    function delay(f, ms) {
                return new Proxy(f, {  apply(target, thisArg, args) {
                                           setTimeout(() => target.apply(thisArg, args), ms);
                                        }
                                    });
                          }

Difference between plain function and proxy ?
Plain Function:
    function sayHi(user) { alert(`Hello, ${user}!`); }
    alert(sayHi.length);  // 1 (function length is the arguments count in its declaration)
    sayHi = delay(sayHi, 1000); // reassign sayHai. argument count change to 0
    sayHi("sridhar");  // Hello, sridhar! <-prints after 1 sec
    alert(sayHi.length);  // 0 (in the wrapper declaration, there are zero arguments)

Proxy :
    function sayHi(user) { alert(`Hello, ${user}!`); }
    alert(sayHi.length);  // 1 (function length is the arguments count in its declaration)
    sayHi = delay(sayHi, 1000); // reassign sayHai. sayHi argument count change to 0
    alert(sayHi.length);  // 1 (proxy forwards "get length" operation to the target)
    sayHi("sridhar");  // Hello, sridhar! <-prints after 1 sec
Note: all operations on the proxy are forwarded to the original function. So sayHi.length is returned correctly after the wrapping in the line (*). We’ve got a “richer” wrapper.

Reflect and Proxy: 
    1. Reflect is a built-in object that simplifies creation of Proxy. 
    2. Reflect allows us to call operators (new, delete…) as functions (Reflect.construct, Reflect.deleteProperty, …).
    3. For every internal method, trappable by Proxy, there’s a corresponding method in Reflect, with the same name and arguments as
         the Proxy trap. 
    4. So we can use Reflect to forward an operation to the original object.

Ex 1:
    let user = {};
    // user.name = 'John'; // or
    Reflect.set(user, 'name', 'John');
    alert(user.name); // John

Ex 2:
    let user = { name: "John" };
    user = new Proxy(user, {
        get(target, prop, receiver) {
            alert(`GET ${prop}`);
            return Reflect.get(target, prop, receiver);
        },
        set(target, prop, val, receiver) {
            alert(`SET ${prop}=${val}`);
            return Reflect.set(target, prop, val, receiver);
        }
    });

    let name = user.name; // shows "GET name"
    user.name = "Pete";   // shows "SET name = Pete"

Proxying a getter:
Let’s see an example that demonstrates why Reflect.get is better. And we’ll also see why get/set have the third argument receiver, that we didn’t use before.
    let user = { _name: "Guest",
         get name() { return this._name; }
     };

    let userProxy = new Proxy(user, {
         get(target, prop, receiver) {
             return target[prop]; // target = user, receiver = admin
             // return receiver[prop]; // Stack will Overflow
         }
    });

    let admin = { __proto__: userProxy,
         _name: "Admin"
     };

    alert(admin.name); // outputs: Guest NOT Admin

    let userProxy = new Proxy(user, {
         get(target, prop, receiver) {
             return Reflect.get(target, prop, receiver); // Admin Admin
             // return Reflect.get(...arguments); // Alternatively
         }
    });

    alert(admin.name); // outputs: Admin

Proxy limitations:
Built-in objects: Internal slots
Many built-in objects ( like Map, Set, Date, Promise etc.,) make use of  “internal slots” for storing data. These slots are accessed directly (not via [[Get]]/[[Set]] internal methods) by those objects. So Proxy can’t intercept that.
    let map = new Map();
    let proxy = new Proxy(map, {});
    proxy.set('test', 1); // Error

Workaround:
    let map = new Map();

    let proxy = new Proxy(map, {
        get(target, prop, receiver) {
            let value = Reflect.get(...arguments);
            return typeof value == 'function' ? value.bind(target) : value;
        }
    });

    proxy.set('test', 1);
    alert(proxy.get('test')); // 1 (works!)
Note: Built-in Array doesn’t use internal slots. That’s for historical reasons. So there’s no such problem when proxying an array.
