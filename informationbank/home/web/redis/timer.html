Lets set a timeout on a value using the EXPIRE command and see what happens. Replace the original code with this:

01.var redis = require('redis')
02., client = redis.createClient();
03. 
04.client.on('error', function (err) {
05.console.log('Error ' + err);
06.});
07. 
08.client.on('connect', runSample);
09. 
10.function runSample() {
11.// Set a value with an expiration
12.client.set('string key', 'Hello World', redis.print);
13.// Expire in 3 seconds
14.client.expire('string key', 3);
15. 
16.// This timer is only to demo the TTL
17.// Runs every second until the timeout
18.// occurs on the value
19.var myTimer = setInterval(function() {
20.client.get('string key', function (err, reply) {
21.if(reply) {
22.console.log('I live: ' + reply.toString());
23.} else {
24.clearTimeout(myTimer);
25.console.log('I expired');
26.client.quit();
27.}
28.});
29.}, 1000);
30.}

Note: Be aware that the timer I use is just to demo the EXPIRE, you should be very careful about using timers in production Nodejs projects.

Run the program. Expected results:

1.Reply: OK
2.I live: Hello World
3.I live: Hello World
4.I live: Hello World
5.I expired
