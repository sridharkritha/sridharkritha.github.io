Set



1. Set<T>:
      1. Access the elements by their key (unlike sequence containers, which access the elements by their relative or absolute position).
      2. Internally, the elements in a set are always sorted from lower to higher.
      3. Typically implemented as binary search trees.
      4. Unique element values. The element value is the key itself.
    #include <iostream>
    #include <set>
    using namespace std;

    int main()
    {
        int myints[] = { 70,70,20,60,40,10 };
        set<int> myset(myints, myints + 5); // 20 40 60 70 (automatically sorted)
        set<int>::iterator it;

        pair<set<int>::iterator, bool> ret;
        ret = myset.insert(30); // 20 30 40 60 70

        // Making use of pair in set
        if (ret.second == false) // if the number already exist
            it = ret.first;      // "it" now points to element 30
        else
            it = ret.first;

        myset.insert(it, 50); // 20 30 40 50 60 70 (automatically sorted)
        for (it = myset.begin(); it != myset.end(); ++it) cout << *it;
    }

2. multiset<T>: Same as set containers, but allowing duplicate keys.
    #include <iostream>
    #include <set>   // same for multiset as well
    using namespace std;

    int main()
    {
        int myints[] = { 70,70,20,60,40,10 };
        multiset<int> myset(myints, myints + 5); // 10 20 40 60 70 70
        multiset<int>::iterator it;

        // returns iterator (NOT pair like "set")
        it = myset.insert(30); // 10 20 30 40 60 70 70
        myset.insert(it, 70);  // 10 20 30 40 60 70 70 70 (iterator just for hint to find the correct position
//  quickly. Eventhough the iterator points to wrong position it ALWAYS insert the at correct pos).

        for (it = myset.begin(); it != myset.end(); ++it) cout << *it << " ";
        cout << myset.count(70); // 3
    }

3. map<K,V>:
    1. Stores data in key and value pairs.
    2. map should not have duplicate keys
    3. key is generally used to identify the value
    4. Elements in the map are sorted from lower to higher key value
    5. map is a telephone guide where the name is the key and the phone number is the mapped value.
    #include <iostream>
    #include <map>
    using namespace std;

    int main()
    {
        map<char, int> mymap;
        map<char, int>::iterator it;
        pair<map<char, int>::iterator, bool> ret; 

        mymap['e'] = 500;
        mymap['a'] = 500;
        // first insert function version (single parameter):
        mymap.insert(pair<char, int>('f', 400));
        mymap.insert(pair<char, int>('d', 300));
        ret = mymap.insert(pair<char, int>('d', 500)); // return pair NOT iterator

        if (ret.second == false)
        {
            cout << "element 'd' already existed";
            cout << " with a value of " << ret.first->second << endl;
        }

        // second insert function version (with hint position iterator):
        it = mymap.begin();
        mymap.insert(it, pair<char, int>('c', 400));
        mymap.insert(it, pair<char, int>('b', 100));

        // showing contents:
        cout << "mymap contains:\n";
        for (it = mymap.begin(); it != mymap.end(); it++) cout << (*it).first << " => " << (*it).second << endl;
    }

Output:     element 'd' already existed with a value of 300

                            b => 100
                            c => 400
                            d => 300
                            e => 500

Example 2: Storing the user-defined data type as a key element for sorting the elements in the map.
    template <class Key, class T, class Compare = less<Key>, class Allocator = allocator<pair<const Key,T> > > class map;
For sorting the primitive data type the map uses std::less<T> as a default 3rd argument but in the case of user-defined data types we should 
         1. explicitly overload the operator <   (or) 
         2. to redefine std::less<T>             (or) 
         3. Use different predicate like greater<T> who should work on the user defined data types.  

    // Storing Stucture as a vector content
    struct student {
            int marks;
            string Name;
            student(int m, string s):marks(m),Name(s) { }
    };

    // Explicitly Overload the operator <
    struct structCompare
    {
        bool operator() (const student& lhs, const student& rhs)
        {
            return lhs.marks < rhs.marks;
        }
    };

    // Define default less<T> using the Template Specialization
    template<>
    struct less<student>
    {
        bool operator() (const student& lhs, const student& rhs)
        {
            // return lhs.marks > rhs.marks; // decending order
            return lhs.marks < rhs.marks;    // ascending order
        }
    };

    // Define default greater<T> using the Template Specialization
    template<>
    struct greater<student>
    {
        bool operator() (const student& lhs, const student& rhs)
        {
            // return lhs.marks > rhs.marks; // decending order
            return lhs.marks < rhs.marks;  // ascending order
        }
    };

    int main()
    {
        // Mehod 1: Define the 3rd default operator less<T> using Template specialization
        map<student,int> mt;
        map<student,int>::iterator itt;

        student s(350,"Lore");
        mt[s] = 12;
        mt[student(400,"Agata")] = 13;   
        mt.insert(pair<student,int>(student(300,"Jesse"),15));

        for(itt=mt.begin();itt!=mt.end();++itt) cout<<(*itt).first.Name<<(*itt).first.marks;


        // Method 2: Explicitly Overload operator <
        map<student,int,structCompare> m;            // NOTE: structCompare
        map<student,int,structCompare>::iterator it; // NOTE: structCompare

        student obj(350,"Lore");
        m[obj] = 12;
        m[student(400,"Agata")] = 13;
        m.insert(pair<student,int>(student(300,"Jesse"),15));   

        for(it=m.begin();it!=m.end();++it) cout<<(*it).first.Name<<(*it).first.marks;

        // Method 3: Use any other predicate instead of less<T>
        map<student,int,greater<student> > mtg;           // NOTE: greater
        map<student,int,greater<student> >::iterator itg; // NOTE: greater
    }

Finding the First Non-repeating Character - Method A

This code is using map.

    Put all characters into a map. If necessary (repeated), it will increment a counter which is the 2nd component of map, value.
    Then, traverse the string and search the map. If found and the counter value is 1, return that character.
    Complexity: traversing (O(N)), searching balanced binary tree (O(log(N)) => O(N + log(N)) => O(N)

    // Returning a character which is the first non-repeated in a string
    #include <iostream>
    #include <string>
    #include <map>

    using namespace std;
    char find(string s)
    {
        map<char,int> myMap;
        int len = s.length();
        for(int i = 0; i < len; i++) ++myMap[s[i]];
        for(int i = 0; i < len; i++) if(myMap.find(s[i])->second == 1) return s[i];
    }

    int main()
    {
        string s="abudabi";
        cout << find(s);
    }

Finding the First Non-repeating Character - Method B

This method is using a simple array utilizing the inversion of character and integer.

Complexity: O(N) - It needs just two traversings.

    // Returning a character which is the first non-repeated in a string
    #include <iostream>
    #include <string>
    using namespace std;

    char find(string s)
    {
        int myArr[256] = {0};
        int len = s.length();
        for(int i = 0; i < len; i++) myArr[static_cast<int>(s[i])]++;
        for(int i = 0; i < len; i++) if(myArr[static_cast<int>(s[i])] == 1) return s[i];
    }

    int main()
    {
        string s="abudabi";
        cout << find(s);
    }
