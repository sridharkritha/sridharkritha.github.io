1. Vector:
              Vector containers are implemented as dynamic arrays. Just as regular arrays, vector containers have their elements stored in contiguous storage locations, which means that their elements can be accessed not only using iterators but also using offsets on regular pointers to elements. But unlike regular arrays, storage in vectors is handled automatically, allowing it to be expanded and contracted as needed.

Vectors are good at:
    * Accessing individual elements by their position index (constant time).
    * Iterating over the elements in any order (linear time).
    * Add and remove elements from the end [NOT from Front] (in constant amortized time).

For operations that involve inserting or removing elements at positions other than the end, they perform worse than deques and lists.

Apart from the "size" function, the vectors also have a "capacity" function, which can be either equal or greater than the actual size. The extra amount of storage allocated is not used but is reserved for the vector to be used in the case it grows.

    int main()
    {
        int ary[] = { 25,50,75 };
        vector<int> v;
        vector<int> another(2, 66); // create a vector with 2 copies of 66
        vector<int>::iterator it;
        it = v.begin();
        it = v.insert(it, 40);  // Insert the element at iterator and old iterator will loss it's value
        it = v.insert(it, 30);  // Needs to use the newly returned iterator instead of old
        v.insert(it, 2, 70);    // Returns "void" and insert 2 copies of 70 (70 70 30 40)
        it = v.begin();         // Bcos old iterator was lost
        v.insert(it + 1, another.begin(), another.end()); // Returns void (70 66 66 70 30 40)  

        v.push_back(50);       // Insert the element at back  (70 66 66 70 30 40 50)

        v.resize(5);           // Drop LAST 2 elements        (70 66 66 70 30)
        v.resize(7, 90);       // Add 2 copies of 90 at LAST  (70 66 66 70 30 90 90)
        v.resize(9);           // Add 2 copies of 0's at LAST (70 66 66 70 30 90 90 0 0)

        v.assign(2, 49);        // Drop ALL the old conents and add the new elements (49 49)
        v.assign(another.begin(), another.end()); // Drop ALL the old conents and add the new elements (66 66)
        v.assign(ary, ary + 2); // Add the array elements without using iterator (25 50) not possible for insert()

        v = another;            // content of another is overwite into v(overloading=) (66 66)
        v.push_back(25);        // v = (66 66 25)

        another.swap(v);        // swapped content: another = [66 66 25], v = [66 66]

        int *p = v.get_allocator().allocate(5);  // allocate an array of 5 elements using vector's allocator        
        for(int i = 0; i < 5; i++) p[i] = i;     // assign some values to array (NOT affect anything in v)
        for(int i = 0; i < 5; i++) cout << p[i]; // 0 1 2 3 4
        v.get_allocator().deallocate(p, 5);

        for (it = v.begin(); it < v.end(); it++) cout << *it << endl;  // 66 66

        cout << "Reverse Iterator" << endl;
        vector<int>::reverse_iterator rit; // Reverse iterator
        for (rit = another.rbegin(); rit < another.rend(); rit++) cout << *rit << endl; // 25 66 66

        return 0;
    }

        vector<int>::iterator it;
        vector<int>::reverse_iterator rit;

        myvector.push_back(30);              // 20 10 30
        myvector.insert(myvector.end(), 5);  // 20 10 30 5 
        myvector.insert(myvector.end(), 40); // 20 10 30 5 40
        myvector[2] = 33;   // Overwriting 30 to 33 already allocated space by
        // push_back / insert
        // myvector[5]= 50;  // Runtime Error - subscript operator can't be used to allocate new space
        vect[3] = 50;        // Here we can use the subscript operator because we delared its size is 5
        vect[1] = 100;

        for (i = 0; i < (int)myvector.size(); i++)                  cout << myvector[i]; // 20 10 33 5  40
        for (rit = myvector.rbegin(); rit < myvector.rend(); ++rit) cout << *rit;  // 40 5  33 10 20
        for (it = vect.begin(); it < vect.end(); it++)              cout << *it;   // 0 100 0 50 0

        myvector.pop_back();                 // Last inserted element will be deleted - 20 10 33 5

        cout << " After pop_back(), now last element is " << myvector.at(i - 2); // 5

        cout << "Size: " << (int)myvector.size() << endl; // 4
        cout << "capacity: " << (int)myvector.capacity(); // 6
        return 0;
    }

Sorting the user-defined data:
    typedef struct data {
        int a;     
string s;
    } data;

    // Method: 1 - Callback function - predicate
    bool comp(const data &l, const data &r) {
        return l.a < r.a;
    }

    // Method 2: Functional Object
    struct great {
        bool operator ()(const data &l, const data &r) { return l.a < r.a; }
    };

    // Method 3: Combining struct declaration and overloading <
    typedef struct pack {
        int a;     
string s;
        // Overloading the operator <
        bool operator <(const pack &d) const { return a < d.a; }

    int main() {
        data d;     
pack p;

        sort(v.begin(), v.end(), great()); // method 2
        for (unsigned int i = 0; i < v.size(); ++i)  cout << v[i].a << v[i].s << endl;
           

        p.a = 10;    p.s = "ten";     vp.push_back(p);

2. deque:
    Double-ended queue has the following properties:
    * Deque manages its elements with a dynamic array, provides random access, and has almost the same interface as a vector.
    * Individual elements can be accessed by their position index.
    * Iteration over the elements can be performed in any order.
    * Elements can be efficiently added and removed from any of their ends (either the beginning or the end of the sequence).
    * A deque frees memory when elements are removed, and it can't happen that elements have to be moved as a result of reallocation in Vector.

Comparison with Vector:
1. Efficient insertion and deletion of elements both at the Beginning and End of the sequence. (Vector Efficient for End of the sequence)
2. deque is not guaranteed to have all its elements in contiguous storage locations, So pointer arithmetics is useless.
3. Unlike vectors, massive reallocations are avoided in deque even if capacity is exhausted by different chunks of storage instead of a block of storage using by vector.
Example:
    int main ()
    {
        int i;
        deque<int> mydeque;
        mydeque.push_back(20); // 20
        mydeque.push_back(10); // 20 10
        mydeque.push_back(30); // 20 10 30
        mydeque.insert(mydeque.end(),5); // 20 10 30 5
        mydeque.insert(mydeque.end(),40); // 20 10 30 5 40
        mydeque[2]= 33; // Overwriting 30 to 33 already allocated space by push_back/insert
       // push_back/insert
        for (i=0; i<mydeque.size(); i++) cout << mydeque[i]; // 20 10 33 5 40

        // erase the 4th element
        mydeque.erase (mydeque.begin()+3);  // 5 will be deleted - NOT ALL elements deleted after iterator

        // erase the first 2 elements:
        mydeque.erase (mydeque.begin(),mydeque.begin()+2); // 20,10 will be deleted

        cout << "mydeque contains:";
        for (i=0; i<mydeque.size(); i++) cout << " " << mydeque[i]; // 33 40
    }

3. List:
  1. List containers are implemented as doubly-linked lists
  2. Stores each in different and unrelated storage locations.
  3. Lack of direct access to the elements by their position
  4. Iterating over the elements in forward or reverse order.
    #include <iostream>
    #include <list>
    using namespace std;

    int main ()
    {
        list<int> mylist (2,400); // two ints with a value of 400
        mylist.push_front (200);
        mylist.push_back (300); // 200 400 400 300
        for (list<int>::iterator it=mylist.begin(); it!=mylist.end(); ++it) cout << *it<< " ";       

        // the iterator constructor can also be used to construct from arrays:
        int myints[] = {16,2,77,29};
        list<int> listConst (myints, myints + 4 );
    
        for (list<int>::iterator it = listConst.begin(); it != listConst.end(); it++)  
            cout << *it << " "; // 16 2 77 29
    }


4. std::array: Lightweight wrapper around a C-style array. But more secured(NO support for implicit conversion to pointer but supports explicit pointer conversion) and also support STL algorithms and features.
 
    std::vector                     Vs       std::array
    ------------------------------------------------------------------------------
    1. template class                        template class 
    2. support STL algorithms                support STL algorithms
    3. grow/sink at run time                 Size fixed at compile time
    4. Dynamic memory(heap)                  Stack memory
    5. heap => slow                          stack => fast
    6. good choice large count of data       good choice for a small count of data

    #include <array> 
    std::array <int> a = {5,8}; // Error size is missing. You can't deduce it from an initializer-list
    std::array <int,4> a={5,8}; // OK a[0] = 5, a[1] = 8, a[2] = 0, a[3] = 0 (initialised by default constructor)

Security Enhancements:
    int* p = a;               // Error, does NOT support implicit conversion to pointer (int*)
    int* p2 = a.data();       // OK - implicit conversion to pointer
    const int* pc = a.data(); // OK - implicit conversion to pointer

NO Array Size declay: Unlike built-in C-style array, which decay into pointer and thus lose track of their size, std::array always knows how many elements it contains:
    void arrSZ(const array<int,5>* parr)
    {
        cout<< parr->size() <<endl; // output 5
    }
