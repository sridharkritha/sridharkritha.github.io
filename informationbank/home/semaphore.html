Semaphore



Windows C++ Producer Consumer Threaded Example:

This Windows C++ example is to demonstrate multi-threading, signalling, semaphores and mutexes using a simple producer and consumer example in C++ on Windows.

This application uses an integer queue. The producer adds values to this queue from 1 to 99 by pushing. Each time it adds a value it notifies a thread via a semaphore that there is something to consume. A consumer thread then consumes this value by popping it.

    #include <Windows.h>
    #include <iostream>
    #include <string.h>
    #include <queue>
    using namespace std;

    DWORD WINAPI threadConsumer(void*);
    DWORD WINAPI threadProducer(void*);

    //Line11:
    #define GETMYRAND() (int)(((double)rand()/(double)RAND_MAX)*100)

    struct MyData {
        queue<int> food;
        bool producerfinished;
        HANDLE mutex;
        HANDLE controlsemaphore;
    };

    int main() {
        MyData mydata;
        mydata.mutex = CreateMutex(NULL, false, NULL);
        mydata.controlsemaphore = CreateSemaphore(NULL,0,2,NULL);
        mydata.producerfinished=false;

        HANDLE handles[3];

        // Lines 28-30:
        handles[0] = CreateThread(0,0,&threadConsumer, (void*)&mydata , 0,0);
        handles[1] = CreateThread(0,0,&threadProducer, (void*)&mydata, 0,0);
        handles[2] = CreateThread(0,0,&threadConsumer, (void*)&mydata, 0,0);

        //Line 32:
        WaitForMultipleObjects(3, handles, true, INFINITE); //"Join" threads

        CloseHandle(mydata.mutex);
        CloseHandle(mydata.controlsemaphore);
    }

    DWORD WINAPI threadProducer(void* lp) {
        MyData * md = (MyData*)lp;

        for(int i =0 ; i < 100; i++)
        {
            char tmps[32];
            sprintf(tmps, "%d\t\t%d\n",i,GetCurrentThreadId());
            cout<<tmps;

            Sleep(GETMYRAND());

            WaitForSingleObject(md->mutex, INFINITE); //Mutex Lock
            md->food.push(i);
            if(md->food.size()>0)
                //Line 73:
                ReleaseSemaphore(md->controlsemaphore,1, NULL); // Release one
                                                                consumer thread
            ReleaseMutex(md->mutex); //Mutex Unlock
        }
        md->producerfinished=true;
        ReleaseSemaphore(md->controlsemaphore,2, NULL); //Release both consumer threads
    }

    DWORD WINAPI threadConsumer(void* lp) {
        MyData * md = (MyData*)lp;

        while(!md->producerfinished)
        {

            Sleep(GETMYRAND());

            //Line 45 :
            WaitForSingleObject(md->controlsemaphore,INFINITE); //Wait for producer

            Lines 47:
            WaitForSingleObject(md->mutex, INFINITE); //Mutext Lock - allow only one
                                                        consumer thread

            while(md->food.size()>0)
            {
                char tmps[32];
                sprintf(tmps, "\t%d\t(%d)\n",md->food.front(), GetCurrentThreadId());
                cout<<tmps;
                md->food.pop();
            }
            // Lines 54:
            ReleaseMutex(md->mutex); //Mutex Unlock
        }
    }

Explanation:

Line11: Randomly generate a sleep time between 0 and 100 milliseconds.

Lines 28-30: Start all the threads giving the same MyData variable to them so the semaphore and the mutex is shared between all of them.

Line 45 : Both consumer threads wait on the semaphore.

Line 73: The producer thread triggers the semaphore allowing one waiting thread to proceed.

Lines 47 & 54: Lock and unlock mutexes allowing only a single consumer thread to run this section of code. This is like a critical section or a 'synchronized' section java.

Line 32: Wait for all the threads to finish processing. Like the join command in Linux.
