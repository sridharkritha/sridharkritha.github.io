assign(): The operator= may be used if the new value can be described by a single parameter; otherwise, the member function assign(), which has multiple parameters, can be used to specify which part of the string is to be assigned to a target string.

        str1 = "Hello";
        str2 = "Agata";
        const char *str3 = "World"; // C - String

        // assigning a specific number of characters of a certain value to a string
        str1.assign(3, '!'); // !!! (fully replace "Hello")
        cout<<endl<<str1;

        // assigning a value from a range of one string to another string
        str1.assign(str2.begin() + 1, str2.end() - 1); // gat (fully replace "!!!")
        cout<<endl<<str1;

        // assigning a number of the characters from one string to another string
        str1.assign(str2, 2, 3); // ata
        cout<<endl<<str1;

        // assigning a number of a C-string characters to a string
        str1.assign(str3, 3); // Wor
        cout<<endl<<str1;

        // assigning the characters of a C-string to a string
        str1.assign(str2); //Agata

Points to be noted
1. Giving pointer to cout, will  prints an address. But if the pointer is type char * then it prints the pointed-to-string. If we want to see the starting address of the string, we should cast the pointer to  int *. Thus, pStr prints the string "hello", but (int *)pStr prints the address.
2. Assigning cstr to pStr does not copy the string, it copies the address. This results in both the  pointers (cstr and pStr) point to the same memory location and string.

    #include <sstream>      // stringstreams
    int main() {
        int num;
        string str;
        stringstream inout; // inout is just like a 'cout' and 'cin' stream

        inout.clear(); // must - for clearing the buffer stream
        cout << num;   // 49 - integer

        cin >> str;    // put the hex number "A5"
        inout << std::hex << str; // put the "A5" hexa string in to the stream buffer
        inout >> num;  // get the 165 as integer from the stream buffer
        inout.clear(); // must - for clearing the buffer stream
        cout << num;   // 165 - integer

        /* or - with error check */
        if (inout >> num)    cout << "successful Conversion";  // Non-zero value returned
        else                 cout << "Problem in Conversion";  // Zero returned
    }

Note:   1. if num data type is double then string convert to double.
            2. All operations applicable to cin and cout is also applicable to inout. i.e., inout.clear(), inout.ignore(),inout.precision(2).

strstr Library Implementation:
    char* strstr(const char *in, const char *str) {
        char c;
        size_t len;

        c = *str++;
        if (!c)
            return (char *) in; // Trivial empty string case

        len = strlen(str);
        do {
            char sc;

            do {
                sc = *in++;
                if (!sc)
                    return (char *) 0;
            } while (sc != c);
        } while (strncmp(in, str, len) != 0);

        return (char *) (in - 1);
    }

0. Trim forward and backward spaces from the string:
// _ _ ab _ c _ d _ _  => ab _ c _d

    char* Trim_C(char* data) {
        size_t len=0,ft=0,bt=0,i=0;
        len = strlen(data);
        for(i=0;i<len;i++)
        {
            if(data[i]==' ')
                ft =ft+1;
            else
                break;
        }
        for(i=len;i>0;i--)
        {
            if(data[i]==' ')
                bt =bt+1;
            else
                break;
        }
        for(i=0;i<(len-ft-bt);i++)
            data[i]=data[i+ft];
        data[i]= '\0';
        return data;
    }

    string Trim_CPP(string str) {
        string temp;
        size_t f=0,b=0;
        for(size_t i=0;i<str.length();++i)
        {
            if(str[i]!=' ')
            {
                f = i;
                break;
            }
        }
        for(size_t i=str.length();i>0;--i)
        {
            if(str[i-1]!= ' ')
            {
                b = i-1;
                break;
            }       
        }

        str.assign(str,f,(b-f)+1);
        return str;
    }
