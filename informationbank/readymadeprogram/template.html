}
else
assert(false);

}

int main()
{
freopen("in.txt","r",stdin);
ofstream fout("out.txt");
int ca;
cin>>ca;
For(cc,1,ca)
{
int a,b,c=0;
cin>>a>>b;
Fill(visited,false);
For(i,a,b)
{
if(visited[i]) continue;
int dig=digs(i);
int num=i;
vi list;
list.push_back(i);
rep(j,dig)
{
int l=num%10;
num/=10;
num+=(int(pow(10.0,dig-1)+0.5))*l;
visited[num]=true;
if(num<=b && num>=a && find(All(list),num)==list.end()) list.push_back(num);
}
c+=(list.size()*(list.size()-1))/2;
}
fout<<"Case #"<<cc<<": "<<c<<endl;
}


return 0;
}
