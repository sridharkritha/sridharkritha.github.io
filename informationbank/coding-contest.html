// Method 3:
// Read a file line by line and word by word
#include <iostream>
#include <string>
#include <fstream> // Reading a file Word by Word.
#include <vector>
 
std::vector<std::string> wordVector;
 
void parseLine(std::string line)
{
	std::string delimiter ="\t=,() "; // Multiple Delimiter - 1. Tabs Space 2. Equal to 3. Comma and 4. Space
	std::size_t prev = 0, pos;
	while ((pos = line.find_first_of(delimiter, prev)) != std::string::npos)
	{
		if (pos > prev)
			wordVector.push_back(line.substr(prev, pos-prev));
		prev = pos+1;
	}
	if (prev < line.length())
		wordVector.push_back(line.substr(prev, std::string::npos));
}
 
int main()
{
	std::ifstream infile("textFile.txt");
	std::string line;
	while (std::getline(infile, line))
	{
		if (line.empty()) continue;
 
		wordVector.clear();
		parseLine(line);
	}
	return 0;
}

17. De Morgans Law and Boolean Algebra:

!( P && Q) isequalto !P || !Q
!( P || Q) isequalto !P && !Q
P || Q     isequalto !(!P && !Q)
P && Q     isequalto !(!P || !Q)

!(P < Q)   isequalto  ( P >= Q )
!(P != Q)  isequalto ( P == Q)
Examples:
!(x!=5 && x!=7)   isequalto    x==5 || x==7
!(x<5 || x>=7)    isequalto    x>=5 && x<7

Note: This equalities may not be true while using float data type in C++.

/*
Operator Precedence:
-------------------
(Left to Right)
------------------------------------------------------
* / % + - < <= > =>    
------------------------------------------------------
==  !=             (Comparison - Reference Level Sym)      
-----------------------------------------------------
& ^ | && ||
-----------------------------------------------------
(Right to Left)
-----------------------------------------------------
= *= /= %= += <<= >>=   (Assignment)
&= ^= |= 
------------------------------------------------------
*/
 
int main()
{
	int a;
	// * / % - Same precedence (  Left to Right )
	a = 5 * 6 % 5;     // 0
	a = 7 * 3 / 5;     // 4  	
	a = 5 + 2 * 7 / 5; // 7
 
	// No need for bracket in case of arithmetic operation. ie., if(P == Q * R)
	int P = 10, Q = 2, R = 5;	
	if(P == Q * R) cout<< "equal"; // Equal
	else cout<< "not equal";
	if(P == R + R) cout<< "equal"; // Equal
	else cout<< "not equal";
 
	// Need for bracket in case of logical and bitwise operations to be done 'first'. if(X == (Y && Z))
	bool X = false, Y = false, Z = true;
	if(X == Y && Z) cout<<"(X == Y) && Z"; // (X == Y) && Z
	else cout <<"X == (Y && Z)";
 
	cout<<X&&Y; // wrongly parsed as: (cout<< X) && Y. Correct form is cout << (X && Y) 
 
	return 0;
}
