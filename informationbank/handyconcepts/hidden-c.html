1. You can do computation even using empty main() !!!
( ref: http://stackoverflow.com/questions/17530187/how-is-it-possible-to-declare-nothing-inside-main-in-c-and-yet-have-a-workin/17530257#17530257 )
Example 1:
    void print_fibs()  {
        //implementation
    }
    
    // Global Variable
    int ignore = (print_fibs(), 0);
    
    int main() {}
In this code, the global variable ignore has to be initialized before entering into main() function. Now in order to initialize the global, print_fibs() needs to be 
executed where you can do anything

Example 2:
    class cls {
    public:
        cls() {
        // Your code for fibonacci series
        }
    } objCls;
    
    int main() { }

Global variable of the class is declared, the constructor is called and there you add the logic to print out the Fibonacci series.

Note that such code is not safe and should be best avoided in general.  For example, the std::cout object may not be initialized when print_fibs() is executed, if so then what would std::cout
do in the function? However, if in other circumstances, it doesn't depend on such initialization order, then it is safe to call initialization functions (which is a common practice 
in C and C++).

2. By default, object in Associate containers are stored as const object (Unlike Vector):
(ref: http://stackoverflow.com/questions/5973427/error-passing-xxx-as-this-argument-of-xxx-discards-qualifiers/5973485#5973485)

class StudentT {
    public: 
        StudentT(int _id) : id(_id) { }
        // OK - Set and Vector 
        int Const_getId() const { return id; }
        // Error - Associative container Set, OK - Vector
        int nonConst_getId()  { return id; }    
        int id;
    };
    
    inline bool operator< (const StudentT& s1, const StudentT& s2) {
        return  s1.Const_getId() < s2.Const_getId();
    }
    
    int main() {
        StudentT s1(1002);
        StudentT s2(1005);
    
        vector<StudentT> vSet;  
        set<StudentT> sSet;
        // VECTOR OBJECT
        vSet.push_back(s1);
        vSet.push_back(s2);
        vector<StudentT>::iterator vItr;
        for (vItr = vSet.begin(); vItr != vSet.end(); ++vItr) {
            cout << vItr->Const_getId()<<endl;    // OK
            cout << vItr->nonConst_getId()<<endl; // OK
        }
        // SET OBJECT
        sSet.insert(s1);
        sSet.insert(s2);
        set<StudentT>::iterator sItr;
        for (sItr = sSet.begin(); sItr != sSet.end(); ++sItr) {
            cout << sItr->Const_getId()<<endl;    // OK
            cout << sItr->nonConst_getId()<<endl; // Error: Bcos objects in the std::set are stored as const StudentT so member function should be const
        }
    
        set<int> iSet;
        iSet.insert(7);
        iSet.insert(5);
        set<int>::iterator isItr;
        for (isItr = iSet.begin(); isItr != iSet.end(); ++isItr) {
            cout << *isItr <<endl;    // OK
        }
    }
3. Declaration order matters for function declaration with Default arguments:
 (ref: http://stackoverflow.com/questions/4989483/where-to-put-default-parameter-value-in-c/24287949#24287949)
        //declaration
        void function(char const *msg, bool three, bool two, bool one = false);
        void function(char const *msg, bool three = true, bool two, bool one); // Error 
        void function(char const *msg, bool three, bool two = true, bool one); // OK
        //void function(char const *msg, bool three = true, bool two, bool one); // OK
    
        int main() {
            function("Using only one Default Argument", false, true);
            function("Using Two Default Arguments", false);
            function("Using Three Default Arguments");
        }
    
        //definition
        void function(char const *msg, bool three, bool two, bool one ) {
            std::cout<<msg<<" "<<three<<" "<<two<<" "<<one<<std::endl;
        }

4. Suppress the compiler warnings for unused variable and unsaved return values or expressions.
    int fn() { return 5; }
    
    void main() {
        (void) fn(); // you know this function returns but you're explicitly ignoring it.
    }

 
Other Possiblities:

    (void)(unusedVar);
    (void)(fn());
    (void)(x &&y || z);
