#include<iostream>
#include<queue>
#include<vector>
#include <algorithm>
 
 
    template<typename T, typename Param>
    class Task
    {
     public:
	   typedef void (T::*Func)(Param&);
 
	    T* object;
	    Func func;
	    Param& param;
		int priority;
 
	    Task(T& obj_, Func f_, Param& p_, int pp)
		     :object(&obj_), func(f_), param(p_),priority(pp) { }
	    virtual ~Task() {}
	    virtual void operator()() { (object->*func)(param); }
	    
        // Copy Constructor -- Not sure... I need to allocate new memory ??
	    Task(const Task& obj)
		       :object(obj.object), func(obj.func), param(obj.param),priority(obj.priority){
		   // object = obj.object; func = obj.func; param=obj.param;
	     }
	    // Copy and Swap Idiom
	    Task& operator=(const Task obj){
			object= obj.object; func = obj.func; param = obj.param;priority = obj.priority;
		     // std::swap(*this,obj);
		      return *this;
	     }
    };
 
// Predicate: Highest Priority Task should be at the top of the queue
 
    struct TaskCompare
    {
	   template<typename T, typename Param>
	   bool operator()(const Task< T,Param > &t1, const Task< T,Param > &t2) const {
		return t1.priority < t2.priority;		
	   }
    };
 
 
    template<typename T, typename Param>
    class taskQ
    {
    public:
	    std::priority_queue<Task< T,Param >, std::vector<Task< T,Param > >, TaskCompare> queue;
 
	    void addTask(T& t, void (T::*f)(Param&), Param& p_, int pri) {
		queue.push(Task< T,Param >( t, f, p_, pri));
	     }
 
	     void executeTask() {	
			 while(queue.size()) {
				 queue.top()();
				 queue.pop();
				}
		
	     }
    };
 
    typedef struct Param_t {
	int value;
    }Param_t;
 
    class Core {
    public:
	Core() {}
	void print(Param_t& p) { std::cout<<p.value;
        }
    };
 
 
    int main(int argc, char ** argv) {
	  taskQ<Core,Param_t&> t;
	  Core *c = new Core();
	  Param_t param;
	  param.value = 3;
	  t.addTask(*c,&Core::print, param,param.value);
	  Param_t param1;
	  param1.value = 1;	 
	  t.addTask(*c,&Core::print, param1,param1.value);
	  Param_t param2;
	  param2.value = 2;
	  t.addTask(*c,&Core::print, param2,param2.value);
	  Param_t param3;
	  param3.value = 4;
	  t.addTask(*c,&Core::print, param3,param3.value);
	  t.executeTask(); 
 
	  return 0;
     }
