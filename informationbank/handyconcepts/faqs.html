1. Deep Copying:
----------------------

#include <iostream>
#include <cstring>
using namespace std;

class MyString {
public:
    MyString() :
            m_str(NULL), m_len(0) {
    }
    ~MyString() {
        if (NULL != m_str)
            delete[] m_str;
    }
    MyString(const char* str) {
        m_len = strlen(str) + 1;
        m_str = new char[m_len];
        if (m_str != NULL) {
            strncpy(m_str, str, m_len);
            m_str[m_len] = '\0';
        } else
            cout << "Memory Allocation Failure";
    }
    MyString(MyString const& obj) {
        m_len = obj.m_len;
        if (obj.m_str) {
            m_str = new char[m_len];
            if (m_str != NULL)
                strncpy(m_str, obj.m_str, m_len);
            else
                cout << "Memory Allocation Failure";
        }
    }

    MyString& operator=(MyString const& obj) {
        if (this == &obj)  // *this = *&obj = obj
            return *this;

        m_len = obj.m_len;
        if (obj.m_str) {
            m_str = new char[m_len];
            if (m_str != NULL)
                strncpy(m_str, obj.m_str, m_len);
            else
                cout << "Memory Allocation Failure";
        }
        return *this;
    }
    void show(MyString const& obj) {
        cout << obj.m_str << endl;
    }

private:
    char* m_str;
    int m_len;
};

int main() {
    // Static Object
    MyString m("sri");
    m.show(m);
    MyString n;
    n = m;
    n.show(n);
    MyString p = n;
    p.show(p);

    // Dynamic Object
    MyString *v = new MyString("Kavitha"); // *v = *this
    v->show(*v); // like *this = Object
    delete v;

    return 0;
}

// Operator Overloading

class prepost {
public:

    prepost(int i = 0) {
        j = i;
    }
    prepost& operator++();
    const prepost operator++(int);
    void show() {
        cout << j << endl;
    }
private:
    int j;
};

prepost& prepost::operator++() {
    j = j + 1;
    return *this;
}
const prepost prepost::operator++(int) {
    prepost temp;
    temp = *this;
    ++*this;
    return temp;
}

int main() {
    prepost a_obj(5);
    prepost b_obj(10);
    a_obj++;
    a_obj.show();
    ++b_obj;
    b_obj.show();
    return 0;
}

#include <sstream>
using namespace std;

class multiply {
private:
    int i, j;
public:
    multiply(int a, int b) :
            i(a), j(b) {
    }

    operator int() {
        return j * i;
    }

    operator string() {
        std::stringstream k;
        k << (i * j);
        return k.str();
    }
};

int main() {
    int i = multiply(7, 6);
    cout << i << endl;
    string str = multiply(7, 6);
    cout << str << endl;
    return 0;
}

// Map: storing structure value as a Key

typedef struct pack {
    int age;
    string name;
} pack;

struct comp {
    bool operator ()(pack const& p, pack const& q) {
        return p.age < q.age;
    }
};

int main() {
    pack p;
    map<pack, int, comp> m;
    map<pack, int, comp>::iterator it;
    p.age = 6;
    p.name = "ravi";
    m[p] = 17;

    p.age = 2;
    p.name = "vijay";
    m[p] = 19;

    p.age = 6;
    p.name = "suresh";
    m[p] = 13;

    for (it = m.begin(); it != m.end(); it++) {
        cout << it->first.age << "  " << it->second << endl;
    }

    return 0;
}

// Virtual Function

class base {
public:
    virtual void show() {
        cout << "Base::show(virtual)" << endl;
    }
    void display() {
        cout << "Base::display" << endl;
    }
    virtual ~base() {
        cout << "Base::virtualDestructor" << endl;
    }
};

class derive: public base {
public:
    virtual void show() {
        cout << "derive::show(virtual)" << endl;
    }
    void display() {
        cout << "derive::display" << endl;
    }
    void dshow(){
        cout<<"dshow\n";
    }
    virtual ~derive() {
        cout << "derive::virtualDestructor" << endl;
    }
};

int main() {
    base *bptr = new derive();
    bptr->show();
    bptr->display();
    ((derive*)bptr)->dshow();

    base &bref = *(new derive());
    bref.show();
    bref.display();
    return 0;
}
