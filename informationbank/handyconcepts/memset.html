        char ch;

Bitfield setting using enum:

    // 1 Nibble(4 bits) - bitfield coding
    typedef enum ballColour_t {
        RED_BALL    = 1 << 0, /* 0th bit */
        BLACK_BALL = 1 << 1, /* 1st bit */
        WHITE_BALL = 1 << 2, /* 2nd bit */
        GREEN_BALL = 1 << 3, /* 3rd bit */
    
    }ballColour_t

    #define isExist(x,y)    ((x & y)==y)

    void main() {
        ballColour_t bagOfBalls = (ballColour_t )13; // 1101 = (8*1 + 4*1 + 2*0 +1) = 13
        cout<< isExist(bagOfBalls ,RED_BALL)  ? "Yes" : "No"; // Prints => YES
        cout<< isExist(bagOfBalls ,BLACK_BALL)? "Yes" : "No"; // Prints => NO
        cout<< isExist(bagOfBalls ,WHITE_BALL)? "Yes" : "No"; // Prints => YES
        cout<< isExist(bagOfBalls ,GREEN_BALL)? "Yes" : "No"; // Prints => YES
    }
