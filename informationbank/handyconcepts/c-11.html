Uniform Initialization / Curly Brace Initialization:
There are 3 types of Initialization using a curly brace. Listed as per its Priority order
1. Initialization_list constructor
2. Constructor that takes the appropriate parameters initializer.
3. Aggregate(class/struct data memeber) initializer.

Note: Uniform Initialization -  Search Order
1. Initialization_list constructor
2. Regular Constructor that takes the appropriate parameters
3. Aggregate initializer
#include <iostream>
#include <vector>
#include <string>
using namespace std;

class dog {
    public:
    // Priority - 3 (Aggregate initializer)
    int age;                                // NOT Called (lower priority)

    // Priority - 2 (Regular Constructor)
    dog(int age) {                         // NOT Called (lower priority)
        this->age = age;
    }

    // Priority - 1 (Initialization_list)
    dog(const initializer_list<int>& v) { // Called and Initialized
        this->age = *(v.begin());
    }
};

int main()
{
    dog d{3};
}

9. Override -  for virtual function
#include <iostream>

////////////////////////////////////////////////////////////////////////////////////////
// override (for virtual function)
// To avoid inadvertently creating a new function in the derived class instead of overriding.
///////////////////////////////////////////////////////////////////////////////////////
// C++ 03 - No check for whether or not function is overrided.
class base {
    virtual void A(int);        // {} - not needed
    virtual void B() const {};  // {} - must
    void C() {};                // {} - must. Also 'void C(int)' - needs {} must.
};

class derived : public base {
    // different param - not overriden. A new function is created
    virtual void A(double)  {}; // No Error
    // const is missing - not overriden. A new function is created
    virtual void B()  {};       // No Error
    // virtual keyword is missing in the base function. A new function is created
    void C() {};                // No Error
};

// C++ 11 - Shows compile time error if there is no proper function override structure  
class base {
    virtual void A(int);        // {} - not needed
    virtual void B() const {};  // {} - must
    void C() {};                // {} - must
};

class derived : public base {   
    virtual void A(double) override {}; // Compile time Error -  no function to override
    virtual void B() override {};       // Compile time Error -  no function to override
    void C() override {};               // Compile time Error -  no function to override
};

int main()
{
    derived d;
    return 0;
}

    // Lvalues: x, i, d1, p, q

    //////////////////////////////////////////////////////////////////////////////////////////////

    // Reference (or) Lvalue Reference
    int k = 2; // k is a Lvalue.
    int& r = k;  // r is a Lvalue Reference. Bcos 'r' reference the Lvalue 'k'.

17. Move Semantics: Move Constructor / Move Assignment Operator

    How did Move Semantics get started?  It was all about optimizing std::vector<T>.

    How do a vector copy and vector move work internally? Vector internally uses a plain array for storing data.

Example:
    sriVector(const sriVector& rhs) // Copy Constructor
    sriVector(sriVector&& rhs)      // Move Constructor

    sriVector& operator=(const sriVector& rhs) // Copy assignment Operator
    sriVector& operator=(sriVector&& rhs)      // Move assignment Operator

Move semantics - Avoids costly and unnecessary deep copying.
    1. Particularly powerful when both passing by value/reference are needed.
    2. Not needed for the case, the function objects are only passed by reference.

    return by value[ return obj; ]:     1. move or 2. copy constructor gets called.
    pass by Lvalue [ foo(obj); ]:       1. ONLY copy constructor gets called.
    pass by Rvalue [ foo(move(obj)); ]: 1. move or 2. copy constructor gets called.

class sriVector {
public:
    int size;
    int* _arr; // A big array

    // Default constructor
    sriVector(int sz) {
        size = sz;
        _arr = new int[size];
        for(int i = 0; i < size; ++i) _arr[i] = i + 99;
    }

// PRIORITY: 'move' constructor > 'copy' constructor
    // Pass by Rvalue (internal temp object)         -  Move Constructor
    // Pass by Lvalue (explicit extra variable used) -  Copy Constructor

    // Copy Constructor - Deep copy
    // Copy constructor is called on 3 instances
    // 1. When instantiating one object and initializing it with values from another object.
    // 2. When passing an object by value.
    // 3. When an object is returned from a function by value.
    sriVector(const sriVector& rhs) {
        size = rhs.size;
        _arr = new int[size];
        for(int i = 0; i < size; ++i)  _arr[i] = rhs._arr[i];
    }
    
    // Move Constructor - inexpensive shallow copy (tempObj become real object)
    sriVector(sriVector&& rhs)  // rvalue reference as the paramenter
    {
        size = rhs.size;
        _arr = rhs._arr;
        rhs._arr = nullptr; // important!!
    }

    // Destructor
    ~sriVector() { 
        delete [] _arr;
    }
};

template<typename T>
void relay(T arg) {

}

// Argument Perfect Forwarding - used to avoid costly and unnecessary copy construction
template<typename T>
void relay(T&& arg) { // T&& - is Universal reference
    // Argument Perfect Forwarding - forwarding the 'arg' from 'relay' to 'foo' without any type change.
    // Rvalue is forwarded as Rvalue and Lvalue is forwarded as Lvalue.
    // relay(reusable)          => arg is Lvalue
    // relay(createSriVector()) => arg is Rvalue

    foo(std::forward<T> (arg)); // perfect forwarding - NO internal Lvalue/Rvalue conversion.
}

// Implementation of std::forward()
template<class T>
T&& forward(typename remove_reference<T>::type& arg) {
    return static_cast<T&&> (arg); // static_cast with return type T&&
}

18. How perfect forwarding works ?
Reference collapsing rule for Type deduction (C++ 11):
    template<typename T>
    void relay(T&& arg) // T&& - is NOT always rValue reference. 
    {  // It is decided by arg(L/R ref) and Ref collapsing rule.
    }

Reference collapsing rule:
1. T& &   => T&
2. T& &&  => T&
3. T&& &  => T&
4. T&& && => T&&

Rvalue to Rvalue reference:
    // T&& variable is initialized with Rvalue => Rvalue reference
    relay(9); => relay(T&&) { }
    9 => int&&  // passing Rvalue
    T&& => int&& && => int&& (Rvalue reference)

Lvalue to Lvalue reference:
    // T&& variable is initialized with Lvalue => Lvalue reference
    relay(x); => relay(T&&) { } // regardless of x is Lvalue of integer or integer reference
    x => int&  // passing Lvalue
    T&& => int& && => int& (Lvalue reference)

Rvalue Reference(T&&):
    1. Rvalue reference is in the form T&&.
    2. T&& - is ALWAYS takes Rvalue reference if T is Non-template type. ex: int&&, float&&
    3. T&& - is Universal reference and can take Rvalue, Lvalue, const, non-const etc., ONLY if T is
function template type(Even NOT a class template type). 
        template<typename T>
        void relay(T&& arg) // T&& - Universal
        {                   
        }

Object: Objects are values. Assignment makes a copy.

C++ 98:

std::vector<T>::push_back(bigObject) -   push_back creates a copy of the bigObject you want to push and stores it in the vector. 

If you want to store pointers to objects in your vector, create a std::vector<whatever*> instead of std::vector<whatever>. However, you need to make sure that the objects referenced by the pointers remain valid while the vector holds a reference to them (smart pointers utilizing the RAII idiom solve the problem).

C++ 11: All the standard containers (std::vector, std::map, etc) support move semantics.

std::vector<T>::push_back(bigObject) - push_back will perform a 'move' instead of a 'copy' if the argument is an 'rvalue reference'.  (Objects can be converted to 'rvalue references' with std::move())

    *

    * std::chrono::steady_clock:  goes at a uniform rate (it can't be adjusted - phone stopWatch)

    std::ratio<2,10>  r; // 1/5

    * std:chrono::duration<>:  represents time duration (ex: 2 hours, 30 minutes, 45 seconds)

    *        nanoseconds, microseconds, milliseconds, seconds, minutes, hours

    chrono::microseconds mi(2745); // 2745 microseconds

    chrono::nanoseconds na = mi;

    // tuple value can be concatenated
    auto t4 = std::tuple_cat( t2, t3 );  // t4 is tuple<int, string, char, string>
    cout << get<3>(t4) << endl;  // "In for a pound" 

    // type traits
    cout << std::tuple_size<decltype(t4)>::value << endl;  // Output: 4 - Size of the 4th tuple data type
    std::tuple_element<1, decltype(t4)>::type dd;  // dd is a string
}

29. Struct Vs Tuple:
// Ref: C++ 11 Library: When to Use Tuple ?
// https://www.youtube.com/watch?v=VFmyZg7zJ3s&list=PL5jc9xFGsL8FWtnZBeTqZBbniyw0uHyaH&index=20
#include <iostream>
#include <string>
#include <map>
#include <tuple>
using namespace std;

// tuple vs struct

tuple<string, int> getNameAge() { 
    return make_tuple("Bob", 34);
}

int main() {
    struct Person { string name; int age; } p;
    tuple<string, int> t;

    cout << p.name << " " << p.age << endl;  // more readable
    cout << get<0>(t) << " " << get<1>(t) << endl; // not readable like struct

    // Then why we need a tuple?
    // As a one-time only structure to transfer a group of data
    string name;
    int age;
    tie(name, age) = getNameAge();

    // Comparison of tuples
    tuple<int, int, int> time1, time2; // hours, minutes, seconds
    if (time1 > time2) 
        cout << " time1 is a later time";

    // Multi index map
    map<tuple<int,char,float>, string> mm;
    mm[make_tuple(2, 'a', 3.4)] = "Faith will move mountains";
    mm.insert(make_pair(make_tuple(12, 'b', 5.3), "Game start"));
    cout << mm[make_tuple(2, 'a', 3.4)]; // Faith will move mountains
    
    // Little trick for swapping variable values
    int a = 1, b = 2, c = 3;
    tie(b, c, a) = make_tuple(a, b, c);
    cout<< a; // 3
}
