function & inheritance => hiding / overloading / overriding
Hiding => Function name should match. NO need to match the params and return type.

class Base {    
public:
    void fun() { cout << "Base - void";  }
    int fun(int i) { cout << "Base"; return i; }
};

class Derived: public Base { // inheritance is MUST
public:
    // using Base::fun; // Base (Method 2: Bringing ALL the functions with name 'fun' to Derived scope)
    char fun(char c) { cout << "Derived"; return c; }
};

int main() {
    Derived d;
    d.fun(5); // Derived (NOT base bcos it is hidden and fun from Derived binds at compile time)
    d.Base::fun(5); // Base (Method 1: explicit referring)
    d.fun(); // Base - void (Bcos of using Base::fun)

    Base b = static_cast<Base>(d); // Method 3: using static_cast - explicit bind but will do SLICING as well
    b.fun(5); // Base - now fun from base binds with 'd' using static_cast
    // Derived dd = static_cast<Derived>(b); // Error: base => derived is NOT allowed in static_cast without pointer

    // How to avoid slicing ? use pointer
    Base* bp = static_cast<Base*>(&d); // NO SLICING
    bp->fun(5); // Base
    Derived* dp = static_cast<Derived*>(bp); // base => derived is allowed bcos bp is pointing to NON sliced (Base+Derived parts).
    dp->fun(5); // Derived
}

    // about what you are doing bcos no run time checks performed for static_cast

    class Base { };
    class Derived1 : public Base { }; 
    class Derived2 : public Base { };
    int main() {
        Derived1 d1;
        Derived1 d2;
        // Safe - upcast
        Base* bp1 = static_cast<Base*>(&d1);
        Base* bp2 = static_cast<Base*>(&d2);

        // Do NOT use static_cast if you are NOT sure what you want from the downcast.
        // mix-match downcast is also allowed in static_cast but it is dangerous
        // it can be avoided by dynamic_cast but it is slow
        Derived1* dp1 = static_cast<Derived1*>(bp2); // bp2 => d1
        Derived2* dp2 = static_cast<Derived2*>(bp1); // bp1 => d2
    }
///////////////////////////////////////////////////////////////////////////////////////////////////
dynamic_cast: only purpose is to solve the above one & only mix-match problem (d1->b1 then b1->d2).
dynamic_cast is used at run-time to find out correct down-cast.
Setup for dynamic_cast to work correctly:
1. Need at least 1 virtual function in base class.
2. if cast fails for pointer type,   it returns null pointer.
3. if cast fails for reference type, it throws an expection - std::bad_cast

Note:
1. it is slow bcos it checks types at runtime using RTTI(run time type information).
2. if you are sure about what you are doing at 'down cast' then use static_cast.

    struct myStruct {
        int x;  // 0-3 bytes
        int y;  // 4-7 bytes

        char c; // 8th byte

                // 2 extra bytes

        cout << *p; // 85904556 = 3rd address value(4 bytes span visibility)

        char* c = reinterpret_cast<char*>(p); // read only first byte of combined 4 bytes
        cout << *c; // A

        cout <<  reinterpret_cast<bool*>(c); // 00D5FA31 = address

void pointer:  void*
1. void* is universal pointer.
2. we can convert any data type pointer to void* (except function pointer, const or volatile).
3. void* can NOT be dereferenced.

    void fun() { }

    int main() {
        int* i = new int{10};
        void* vi = static_cast<void*>(i); // OK

        cout <<*(static_cast<int*>(vi)); // OK 
        

        const int* ic = new int{10};
        void* vic = static_cast<void*>(ic); // Error - NOT allowed - bcos of const

        volatile int* iv = new int{10};
        void* viv = static_cast<void*>(iv); // Error - NOT allowed - bcos of volatile

        void* vf = static_cast<void*>(fun); // Error - NOT allowed - bcos of function pointer
    }

const_cast:
const_cast<T>: Removes const or volatile qualifiers from pointers or reference variable
T - must be pointer or reference
volatile - tells the compiler to optimise the variable bcos it will frequently changes.

    // 1. Use const_cast only if the actually refered object/variable is NOT const
    int main() {
        // Do NOT use const_cast
        const int a = 5; // source variable is const so You can NOT change the constness even using const_cast
        const int* p = &a;
        int* q = const_cast<int*>(p); // removing constness of 'p' NOT 'a'
        *q = 9;     // NO Error but INVALID or Undefined

        // Use const_cast
        int i = 5; // you can change
        const int* pi = &i;
        int* qi = const_cast<int*>(pi); // Correct way
        *qi = 9; // VALID - it is stored inside 'i'
        cout << i; // 9
        cout << *qi; // 9

reference Vs pointer:
Once a reference is established to a variable, you cannot change the reference to reference another variable.
referencing and dereferencing are done on the references implicitly. NO need for explicit dereferencing like pointers.    int main()
    {
        int  i = 5;  // &i = 0xFFAA
        int& r = i;  // r = 0xFFAA - reference has NO own address it use already existing address of i
        int* p = &i; // p = 0x9E7E (p has own address(0x9E7E) and it contains i's address(0xFFAA) as it's data)

        // Reassingments
        int x = 9;

        p = &x;    // now p refers to x NOT i (you can reassign in pointer and NOT in reference)
        cout << i; // 5 untoched bcos p & i are disconnected

        // permenantly 'r' binds to 'i', value can be updated but 'r' can NOT refers any other variable.
        r = x;     // OK - value gets update but it still reference to i NOT x;
        cout << i; // 9 NOT 5

        x = 77;
        cout << i; // 9 NOT 77

        r = x;     // OK - value gets update but it still reference to i NOT x;
        cout << i; // 77 NOT 9

        // Arithmetic operation
        p++; // moves to next byte
        // r++; // NOT allowed


        // indirection 
        int** p2 = &p;
        int*** p3 = &p2; // multi levels indirection are allowed for pointer

        int& r2 = r;    // only ONE level indirection is allowed for reference. Both refers the same.
    }
///////////////////////////////////////////////////////////////////////////////////////////////////
When which should be used ?
Reference: Use refernce in function parameters and return types.
1. Pass some BIG objects.
2. To avoid object slicing.
3. To achieve run time polymorphism in a function.
4. To modify local variable of caller function.
Note: In copy constructor ONLY reference can be used(You can NOT use pointer instread).
Compiler give error if do NOT give reference on the copy constructor - Good you will NOT get stack overflow at runtime.

Pointers: Use pointers in data structures like linked list, tree, graph etc.,
1. Sometimes we put NULL/nullptr in node.
2. Sometimes we change pointers to point some another node.

///////////////////////////////////////////////////////////////////////////////////////////////////
symbol * has different meaning in a declaration statement and in an expression. 
When it is used in a declaration (e.g., int * pNumber), it denotes that the name followed is a pointer variable. 
Whereas when it is used in a expression (e.g., *pNumber = 99; temp << *pNumber;), it refers to the value pointed to by the pointer variable.
///////////////////////////////////////////////////////////////////////////////////////////////////
symbol & is different in an expression and in a declaration. When it is used in an expression, & denotes the address-of operator, 
which returns the address of a variable, e.g., if number is an int variable, &number returns the address of the variable number (this has been described in the above section).

Howeve, when & is used in a declaration (including function formal parameters), it is part of the 
type identifier and is used to declare a reference variable (or reference or alias or alternate name). 
It is used to provide another name, or another reference, or alias to an existing variable.

Since swap() takes its parameters by reference they cannot bind to those temporary values. 
using ++x and ++y work as we first increment a and b and then pass it to swap so there is no temporary.

vector size() and capacity():
size() of a vector is the #n of elements that it contains currently.

capacity() is the amount of total space that the vector has. Under the hood, a vector just uses an 'array'. 
The capacity of the vector is the size of that array. This is always equal to or larger than the size. 
The difference between them is the number of elements that you can add to the vector before the array 
under the hood needs to be reallocated.

Size is not allowed to differ between multiple compilers. But capacity size and reallocation are compiler dependent.

    int main() {
        vector<int> v;
        cout << v.size() << v.capacity(); // 0 0
        v.push_back(1);
        cout << v.size() << v.capacity(); // 1 1 (Reallocation 0->1)
        v.push_back(1);
        cout << v.size() << v.capacity(); // 2 2 (Reallocation happens 1->2)
        v.push_back(1);
        cout << v.size() << v.capacity(); // 3 4 (or 3 compiler dependent. Reallocation happens 2->4 )
        v.push_back(1);
        cout << v.size() << v.capacity(); // 4 4 (NO - Reallocation happens)
        v.push_back(1);
        cout << v.size() << v.capacity(); // 5 8 (or 5 compiler dependent 4->8 )
    }

2 types of "new/delete" in C++

1. new/delete operators.
2. new expressions (There is NO delete expression) (or) placement new

new operator[ new T() ]            => 1. Allocate plain memory
                                      2. Calls 'constructor' for - Construct an object in that memory
delete operator[ delete obj ]      => 1. Calls 'destructor'
                                      2. Deallocate plain memory

new expression[ new(address) T() ] => 1. Calls new 'operator' and instruct at which address to create memory allocation and then construct an object   
NO delete expression               => you have to call the destructor manually.

new operator:
1. find the correct size memeory
2. construct an object in that memory.

Placement new expression:
1. Use/mark the memory from the given address upto the size of an object. Note: Plain Memory pool already created by the new operator. 
2. construct an object in the allocated memory location.

Note: There is no "placement-delete": Instead, you have to call the destructor manually.

    Base* b2 = new Base; // Constructor
    Base* b3 = new Base; // Constructor

    char* cp = new char[sizeof(Base) * 3]; // NO call to constructor - just plain memory allocation(memory pool)
    Base* pn1 = new(&cp[0]) Base();        // Constructor : constructs an object - Calls constructor
    Base* pn2 = new(&cp[4]) Base();        // Constructor : Kernal NO need to search for best memeory location
    Base* pn3 = new(&cp[8]) Base();        // Constructor 
    
    // You MUST call the desturctor explicitly
    pn1->~Base(); // destructor. Your responsibility
    pn2->~Base(); // destructor. Do NOT use => delete pn2; - gives undefined behaviour
    pn3->~Base(); // destructor:
    delete[] cp;  // NO destructor called for plain memory deallocation

        const int ci; // 1. const data member - needs initializer list
        int& ref;     // 2. ref   data member - needs initializer list
        int x;        // 3. same variable usage - needs initializer list
    public:
        Base(int x) : ci{ x }, ref{ x }, x{ x }
        {
            // ci = x;  // NOT allowed here
            // ref = x; // NOT allowed here
            x = x;      // 5 = 5 NOT this->x = 5
        }
    };

    class A {
    public:
        A(Base obj) // calls copyconstructor and NO need for initializer list for 'Base obj'
        {
        }
    };

    class E { 
        // 4. Composition of object of a class without a default constructor - needs initializer list.
        Base obj; // Composition: NOTE: Base class does NOT have default constructor so initializer list is NEEDED!
    public:
        E() : obj(4) // without 'obj(4)' intializer list will give compilation error
        {
            // obj = 4; // NOT allowed
        }
    };

    class Derived : public Base {  // 5. Inheritance of a class without a default constructor - needs initializer list.
    public:
        Derived() :Base(1) //  without 'Base(1)' intializer list will give compilation error
        {
        }
    };

    int main() {
        Base b(5); // b = {ci=5 ref=5 }
        A a(b);    // a = {}; b = {ci=5 ref=1701784 } r holds garbage now!!
        E e;       // e = {obj={ci=4 ref=4 } }
        Derived d;
    }

    { 
        // b = x;   // Method 2:
        cout << "A::ParameterConstructor "; 
    }

    A& operator=(const A& obj) { cout << "A::Assignment "; }

};

    Base obj(5); // Base::ParameterConstructor

    // Method 1: without Initialisation list(#4 calls)
    A a(obj);    // 1. Base::CopyConstructor  2. Base::DefaultConstructor  3. Base::Assignment 4. A::ParameterConstructor

                //                              A::ParameterConstructor

What does printf and scanf returns ?
printf => returns #no.of characters printed successfully in the console.
scanf  => returns #no.of elements read successfully from the console.

    #pragma warning(disable:4996) // To disable scanf/printf warning of Error C4996 /(or) use _CRT_SECURE_NO_WARNINGS.

    #include<stdio.h>
    int main() {
        int i;
        char arr[100]; int val;
        int nCharRead = scanf("%i", &i); // 1
        nCharRead = scanf("%i %i %i", &i, &i, &i); // 3 (3 items read %i %i %i)
        nCharRead = scanf("%s", arr); // 1 (NOT no.of items inside array )
        nCharRead = scanf("%s %i", arr, &i); // 2
        int nCharPrinted = printf("%s", "a234e"); // 5
        
        cout << nCharPrinted << nCharRead << val; // 5 2
    }

What is RVO and NRVO ?
RVO  => Return Value Optimisation => unnecessay copyConstructor/copyAssigment calls are avoid on function return and assignements
NRVO => Named RVO

    class Base {
    public:
        Base() { cout << "Default "; }
        Base(const Base&) { cout << "Copy "; }
        Base& operator=(const Base&) { cout << "Assigment "; return *this; }

    };

        Base bb;

    }

        return Base(); // RVO happens and assignemt statement at main
    }

        Base bb = bar(); // Default RVO => bar Default Destructor
                        // without RVO => bar Default Copy Destructor Copy Destructor
    }

Chain Assignment:
    int main() {
        int a = 1, b = 2, c = 3;
        a = b = c; // 'c' is assigned to both 'a' & 'b' 
        cout << a << b << c; // 3 3 3

        a = 1, b = 2, c = 3;
        (a = b) = c; // 'b' is assign to 'a' and disappear but 'a' stay gets the next assignment from 'c'
        cout << a << b << c; // 3 2 3
    }

Use of pointers in local scope are NOT exception safe:

    void f(int n) {
        Gadget* p = new Gadget{n}; // naked pointer in local scope (Use smart pointer instead)
        //...
        if(n > 100) return; // early return - memory leaks!! 
        //..
        delete p; // Never called after above early return. Also C++ NOT have garbage collector !!
    }

Best Solution: Do NOT use any form of pointers(inc. smart pointers). Just use constructor & Destructors for memory acquire/release.
    void f(int n) {
        Gadget g{n}; // constructor will acquire a resource
        //...
        if(n > 100) return; // early return - NO memory leaks!! 
    } // destructor will deallocated the acquired resource

When to use pointers ?
- for iterators, references, array - use pointers
- to reference resources (use pointers from within a handle)
- to pass large amounts of data into a function (Eg. Pass by const reference)

- to return large amount of data out of a function(Do NOT use pointer, use move operation).
- to represent ownership (Do NOT use pointer, use handles like vector, string etc.,)
