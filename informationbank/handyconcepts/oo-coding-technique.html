OO Coding Technique 



1. Auto Delete idiom using RAII:

//  Private copy constructor and copy assignment ensure classes derived 
//  from class NonCopyable cannot be copied.
class NonCopyable 
{
   NonCopyable (NonCopyable const& ); // private copy constructor
   NonCopyable & operator = (NonCopyable const& ); // private assignment operator
};

template <class T>
class AutoDelete : NonCopyable
{
  public:
    AutoDelete (T * p = 0) : ptr_(p) {}
   ~AutoDelete () throw() { delete ptr_; } 
  private:
    T *ptr_;
};

void foo ()
{
  X * p = new X;
  AutoDelete<X> safe_del(p); // Memory will not leak
  if (...)
    if (...)
      return; 
 
  // No need to call delete here.
  // Destructor of safe_del will delete memory
}

2. Scoped Lock idiom using RAII:

class Ilock  // lock interface
{
public:
	virtual ~Ilock() {};
	virtual int Lock() = 0;
	virtual int UnLock() = 0;
};


class CWin32CritSec : public Ilock  /* Ilock  interface - useful for taking advantage of scope lock facility */
{

public:
	CWin32CritSec() { ::InitializeCriticalSection(&m_CritSec); }
	virtual ~CWin32CritSec() { ::DeleteCriticalSection(&m_CritSec); }
	int Lock() { ::EnterCriticalSection(&m_CritSec); return 0; }
	int UnLock() { ::LeaveCriticalSection(&m_CritSec); return 0; }

protected:
	::CRITICAL_SECTION m_CritSec;

private:
	// NonCopyable
	CWin32CritSec(CWin32CritSec const&);
	CWin32CritSec const& operator= (CWin32CritSec const&);
};

// Uses RAII concept for doing auto lock and unlock with respect to function scope
class CscopeLock
{
public:
	// This function is capable of taking CWin32CritSec object as a argument because of oo property
	CscopeLock(Ilock  *pLock):m_pLock(pLock) { m_pLock->Lock(); }
	~CscopeLock() { m_pLock->UnLock(); }
protected:
	Ilock  *m_pLock;
private:
	// NonCopyable
	CscopeLock(CscopeLock const&);
	CscopeLock const& operator= (CscopeLock const&);
};

CWin32CritSec funLock;

void X::bar()
{
  CscopeLock Lock(&funLock); // Lock will be released certainly
   if (...)
    if (...)
      throw "ERROR"; 
  // No need to call release here.
  // Destructor of safe_lock will release the lock
}

3. Scope Guard:

Resource Acquisition is Initialization (RAII) idiom allows us to acquire resources in the constructor and release them in the destructor when scope ends successfully or due to an exception. 
It will always release resources. This is not very flexible. Sometime we don't want to release resources if no exception is thrown but we do want to release them if exception is thrown.
  • To ensure that resources are always released in face of an exception but not while returning normally
  • To provide basic exception safety guarantee


class ScopeGuard
{
public:
  ScopeGuard () 
   : engaged_ (true) 
  { /* Acquire resources here. */ }
 
  ~ScopeGuard ()  
  { 
    if (engaged_) 
     { /* Release resources here. */} 
  }
  void release () 
  { 
     engaged_ = false; 
     /* Resources no longer be released */ 
  }
private:
  bool engaged_;
};
void some_init_function ()
{
  ScopeGuard guard;
  // ...... Something may throw here. If it does we release resources.
  guard.release (); // Resources will not be released in normal execution.
}
