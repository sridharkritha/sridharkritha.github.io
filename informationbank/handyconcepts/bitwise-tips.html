bitwise Tips



Following code snippet will demonstrate usefulness of bitwise operations:

Example 1:

    #include <iostream>
    #include <bitset> // ex: bitset<8>
    #include <limits> // ex: numeric_limits<int>::max()
    using namespace std;
    int main() {
        const unsigned char option1 = 0x01; // hex for 0000 0001 = 2^0
        const unsigned char option2 = 0x02; // hex for 0000 0010 = 2^1
        const unsigned char option3 = 0x04; // hex for 0000 0100 = 2^2
        const unsigned char option4 = 0x08; // hex for 0000 1000 = 2^3
        const unsigned char option5 = 0x10; // hex for 0001 0000 = 2^4
        const unsigned char option6 = 0x20; // hex for 0010 0000 = 2^5
        const unsigned char option7 = 0x40; // hex for 0100 0000 = 2^6
        const unsigned char option8 = 0x80; // hex for 1000 0000 = 2^7

        const unsigned char oneByteMaxValue = 0xFF; // one byte
        const unsigned int fourByteMaxValue = 0xFFFFFFFF; // Four Byte

        // byte-size value to hold some combination of the above 8 options
        unsigned char myflags = 0; // we'll initialize it to all flags turned off to start

        //To turn a bit on, we use bitwise OR:
        myflags |= option4; // turn option 4 on.
        myflags |= option4 | option5; // turn options 4 and 5 on.

        // To turn a bit off, we use bitwise AND with an inverse bit pattern:
        myflags &= ~option4; // turn option 4 off
        myflags &= ~option4 & ~option5; // turn option4 and 5 off.

        //To toggle a bit state, we use bitwise XOR:
        myflags ^= option4; // flip option4 from on to off, or vice versa
        myflags ^= option4 | option5; // flip options 4 and 5

        //To query a bit state, we use bitwise AND:
        if (myflags & option4) // if option4 is set, do something
        {
        }
        cout<<"------------------------------(XOR application)----------------------------------------------------------"<<endl;
        // if (x==a) x=b; (or) if (x==b) x=a (or) x = x^ a ^ b
        int x = 5, a = 5, b = 9;
        x = x ^ a ^ b;
        cout<< x << endl; // 9

        cout<<"------------------------------SET BIT--------------------------------------------------------------------"<<endl;
        a = 5,b = 5;
        cout<<bitset<8>(a)<<endl; // Print in Binary - 0000 0101
        int setBitPos = 7; // note: this is not 8th location
        b |= 1<<setBitPos;
        cout<<bitset<8>(b)<<endl; // 1000 0101

        cout<<"------------------------------CLEAR BIT------------------------------------------------------------------"<<endl;
        int clearBitPos = 7; // Assign zero
        b &= ~(1<<clearBitPos);
        cout<<bitset<8>(b)<<endl; // 0000 0101

        cout<<"------------------------------CHECK BIT------------------------------------------------------------------"<<endl;
        int checkBitSetPos = 2;
        if(a&(1<<checkBitSetPos))
            cout<<bitset<8>(a&(1<<checkBitSetPos))<<" = True"<<endl; // 0000 0100 = True
        else
            cout<<bitset<8>(a&(1<<checkBitSetPos))<<" = False"<<endl;

        cout<<"------------------------------DIVISION AND POWER---------------------------------------------------------"<<endl;
        cout<<(12>>1)<<endl; // 6 = 12/2
        cout<<(16>>3)<<endl; // 2 = 16 / (2^3)

        cout<<"------------------------------MULTIPLICATION AND POWER---------------------------------------------------"<<endl;
        cout<< (3<<1) <<endl; // 6 = 3 * 2
        cout<< (3<<4) <<endl; // 48 = 3 * (2^4)
        cout<< (2<<4) <<endl; // 2^5 = 32; Calculate 2^n by 2<<(n-1)
        cout<<"---------------------------------------------------------------------------------------------------------"<<endl;

        cout<<"Signed Integer Range - Max"<<endl;
        int maxInt = ( 1 << 31) - 1; // (1* 2^31) - 1 (or) (1* 2^(32-1)) - 1
        cout<< maxInt<<endl; // 2,147,483,647
        cout<<hex<< maxInt<<endl; // 0x7FFFFFF
        cout<<dec<< numeric_limits<int>::max()<<endl;
        cout<< sizeof(int)<<" Bytes"<<endl; // 4 bytes
        maxInt = 0x7FFFFFFF;
        cout<<hex<<maxInt<<" = "<<dec<<maxInt<<" this is max value for the 4 byte"<<endl;
        cout<<"Signed Integer Range - Min"<<endl;
        int minInt = 1 << 31; // -(1* 2^31) (or) -(1* 2^(32-1))
        cout << minInt<<endl; // –2,147,483,648
        cout<<hex<< minInt<<endl; // 0x80000000
        cout<<dec<< numeric_limits<int>::min()<<endl;
        cout<< sizeof(int)<<" Bytes"<<endl;
        minInt = 0x80000000;
        cout<<hex<<minInt<<" = "<<dec<<minInt<<" this is min value for the 4 byte"<<endl;
        cout<<"---------------------------------------------------------------------------------------------------------"<<endl;

        cout<<"Unsigned Integer Range - Max"<<endl;
        unsigned int maxUnsignInt = ( 1LL << 32) - 1; // (1* 2^32) - 1; Note: 1<<32 will overflow bcos 1 in 4byte representation so change into 1LL <<32 (here 1 in 8byte representation)
        cout<< maxUnsignInt<<endl; // 4,294,967,295
        cout<<hex<< maxUnsignInt<<endl; // 0xFFFFFFFF;
        cout<<dec<< numeric_limits<unsigned int>::max()<<endl;
        cout<< sizeof(unsigned int)<<" Bytes"<<endl;
        maxUnsignInt = 0xFFFFFFFF;
        cout<<hex<<maxUnsignInt<<" = "<<dec<<maxUnsignInt<<" this is max value for the 4 byte"<<endl;
        cout<<"---------------------------------------------------------------------------------------------------------"<<endl;

        cout<<"Signed long long Range - Max"<<endl;
        long long maxLongLong = ( 1LL << 63) - 1; // (1* 2^63) - 1 (or) (1* 2^(64-1)) - 1
        cout<< maxLongLong<<endl; // 9,223,372,036,854,775,807
        cout<<hex<< maxLongLong<<endl;
        cout<<dec<< numeric_limits<long long>::max()<<endl;
        cout<< sizeof(long long)<<" Bytes"<<endl;
        maxLongLong = 0x7FFFFFFFFFFFFFFF;
        cout<<hex<<maxLongLong<<" = "<<dec<<maxLongLong<<" this is max value for the 8 byte"<<endl;
        cout<<"Signed long long Range - Min"<<endl;
        long long minLongLong = 1LL << 63; // (1* 2^31) - 1 (or) (1* 2^(32-1)) - 1
        cout<< minLongLong<<endl; // 9,223,372,036,854,775,808
        cout<<hex<< minLongLong<<endl; // 0x8000000000000000;
        cout<<dec<< numeric_limits<long long>::min()<<endl;
        cout<< sizeof(long long)<<" Bytes"<<endl;
        minLongLong = 0x8000000000000000;
        cout<<hex<<minLongLong<<" = "<<dec<<minLongLong<<" this is min value for the 8 byte"<<endl;
        cout<<"---------------------------------------------------------------------------------------------------------"<<endl;
        
        cout<<"Unsigned long long Range - Max"<<endl;
        unsigned long long maxUnsignLongLong = ( 1ULL << 64) - 1; // (1* 2^64) - 1 (or) (1* 2^64) - 1 ; Note: 1ULL
        cout<< maxUnsignLongLong<<endl; // 18,446,744,073,709,551,615
        cout<<hex<< maxUnsignLongLong<<endl; // 0xFFFFFFFFFFFFFFFF;
        cout<<dec<< numeric_limits<unsigned long long>::max()<<endl;
        cout<< sizeof(unsigned long long)<<" Bytes"<<endl;
        maxUnsignLongLong = 0xFFFFFFFFFFFFFFFF;
        cout<<hex<<maxUnsignLongLong<<" = "<<dec<<maxUnsignLongLong<<" this is max value for the 8 byte"<<endl;
    }


Example 2:

// 32 bit RGBA = 8bit(255) R + 8bit(255) G + 8bit(255) B + 8bit(255) A

    #include <iostream>
    int main() {
        // Note: in C++11, it's better to use uint8_t than "unsigned char"

        using namespace std;
        cout << "Enter a 32-bit RGBA color value in hexadecimal (e.g. FF7F3300): ";
        unsigned int pixel;
        cin >> hex >> pixel; // std::hex allows us to read in a hex value

        // use bitwise AND to isolate red pixels, then right shift the value into the range 0-255
        unsigned char red = (pixel & 0xFF000000) >> 24;
        unsigned char green = (pixel & 0x00FF0000) >> 16;
        unsigned char blue = (pixel & 0x0000FF00) >> 8;
        unsigned char alpha = pixel & 0x000000FF;

        cout << "Your color contains:\n";
        cout << static_cast<int>(red) << " of 255 red\n";
        cout << static_cast<int>(green) << " of 255 green\n";
        cout << static_cast<int>(blue) << " of 255 blue\n";
        cout << static_cast<int>(alpha) << " of 255 alpha\n";
    }

    Output:
                Enter a 32-bit RGBA color value in hexadecimal (e.g. FF7F3300): FF7F3300
                    Your color contains:
                 255 of 255 red
                    127 of 255 green
                    51 of 255 blue
                    0 of 255 alpha
