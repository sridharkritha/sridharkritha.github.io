      int x;

      char name[10];

 }sv,*sp, svi={5,"hai"};   

 

main()

{

 

struct sri sv;

struct sri *sp;  //Note : svi declaration is not needed

 

scanf("%d",&sv.x);     // Structure Variable type

scanf("%s",sv.name);

printf("%d",sv.x);

printf("%s",sv.name);

 

sv.x=99;                 // Structure Variable type, inside initialization

strcpy(sv.name,"kavi");

printf("%d",sv.x);

printf("%s",sv.name);

 

scanf("%d",&sp->x);  // Structure Variable Pointer

scanf("%s",sp->name);

printf("%d",sp->x);

printf("%s",sp->name);

 

printf("%d",svi.x);     // Structure Variable type, already initialized

printf("%s",svi.name);

 

 

}

169.  Various Data type - Sizeof

main()

{

 

char ch,*ch_ptr,ch_arr[10];

int  in,*in_ptr,in_arr[10];

float flo,*flo_ptr,flo_arr[10];

double dou,*dou_ptr,dou_arr[10];

 

ch='a';

ch_ptr="raman";

strcpy(ch_arr,"kav");

                       

printf("\n%d",sizeof(ch));           // 1

printf("\n%d",sizeof(*ch_ptr));   // 1

printf("\n%d",sizeof(ch_ptr));    // 2

printf("\n%d",sizeof(ch_arr));    // 10

 

in=5;

in_ptr=&in;

in_arr[0]=8;

in_arr[1]=9;

printf("\n\n%d",sizeof(in));       // 2

printf("\n%d",sizeof(*in_ptr));  // 2

printf("\n%d",sizeof(in_ptr));   // 2

printf("\n%d",sizeof(in_arr));  // 20

 

flo=5.5; 

flo_ptr=&flo;

flo_arr[0]=8.8;

flo_arr[1]=9.9;

printf("\n\n%d",sizeof(flo));      // 4

printf("\n%d",sizeof(*flo_ptr)); // 4

printf("\n%d",sizeof(flo_ptr));  // 2

printf("\n%d",sizeof(flo_arr));  // 40

 

dou=5.5;

dou_ptr=&dou;

dou_arr[0]=8.8;

dou_arr[1]=9.9;

printf("\n\n%d",sizeof(dou));     // 8  

printf("\n%d",sizeof(*dou_ptr)); // 8

printf("\n%d",sizeof(dou_ptr));  // 2

printf("\n%d",sizeof(dou_arr));  // 80  

 

}
