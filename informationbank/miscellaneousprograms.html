Using malloc [C]:
#include <stdlib.h> // for malloc
int main()
{
    int **ptr;
    int ROW,COL,i,j;
    printf("\nEnter the no.of ROW & COL:\n");
    scanf("%d%d",&ROW,&COL);

    ptr=(int **) malloc(sizeof(int *) * ROW );
    for(i=0 ; i< ROW; i ++)
        ptr[i]= (int * ) malloc(sizeof(int) * COL);

    printf("\nEnter the Values:\n");
    for(i=0;i<ROW;i++)
        for(j=0;j<COL;j++)
            scanf("%d",&ptr[i][j]); // Note :' & ' symbol

    printf("\nOutput:\n");
    for(i=0;i<ROW;i++)
    {
        printf("\n");
        for(j=0;j<COL;j++)
            printf("%d\t",ptr[i][j]);
    }
}

6.Finding the machine type (8 bit, 16 bit, 32 bit, 64 bit)

Using Left shift Operator :

int main()
{
    long i=0,j=0,n=4;
    printf("\n Compiler Dependant:");
    printf("\nVC++6 Compiler : Size of Int :%d",sizeof(int)); /* 16 bits */
    printf("\nVC++6 Compiler : Size of long : %ld",sizeof(long)); /* 16 bits */
    printf("\n Machine Dependant:");
    while(1)
    {
        i= (2<<(n-2))-1;
        j= 2<<(n-2);
        if(i > j)
        {
            printf("\n %d Bit Machine",n);
            break;
        }
        n=n<<1;
    }
}

Using Pow operator :

int main()
{
    long i=0,j=0;
    printf("\n Compiler Dependant:");
    printf("\n Size of Int :%d",sizeof(int)); /* 16 bits - VC6 */
    printf("\nSize of long : %ld",sizeof(long));/*16 bits-VC6 */

    printf("\n Machine Dependant:\n");
    i= pow(2,7)-1; /* 127 */
    j= pow(2,7);   /* 128 */

    if((i!=0) && (j!=0) && (i >= j))
        printf("8 Bit Machine Type (not compiler dependent) ");

    i= pow(2,15)-1; /* 32767 */
    j= pow(2,15); /* 32768 */

    if((i!=0) && (j!=0) && (i >= j))
        printf("16 Bit Machine Type (not compiler dependent) ");

    i= pow(2,31)-1; /* 2147483647 */
    j= pow(2,31); /* - 2147483648 */

    if((i!=0) && (j!=0) && (i >= j))
        printf("32 Bit Machine Type (not compiler dependent) ");

    i= pow(2,63)-1; /* 0 (some compiler gives overflow error)*/
    j= pow(2,63); /* 0 (some compiler gives overflow error)*/

    if((i!=0) && (j!=0) && (i >= j))
        printf("64 Bit Machine Type (not compiler dependent)");
}

8. Determine whether a machine's byte order is big-endian or little-endian

Method 1:

Introduction:
Consider a 32 bit value: A = 0xABCDEFGH
Big Endian format of storage:       Little Endian format of storage
    BaseAddr+0 -> AB                                     BaseAddr+0 -> GH
    BaseAddr+1 -> CD                                    BaseAddr+0 -> EF
    BaseAddr+2 -> EF                                     BaseAddr+0 -> CD
    BaseAddr+3 -> GH                                    BaseAddr+0 -> AB

Note: Hexa Decimal Representation:
      Hexa     Binary     Decimal
      0x01 => 0000:0001 => 1 
      0x04 => 0000:0100 => 8
      0x40 => 0100:0000 => 64
The main reason for using Hexa representation is short and easy to understand the bit patterns.

Note 2: Endianess will be useful for only storing the values in the memory. Once a value loaded from the memory where it is stored as either in little or big endian format into the processor  will be considered as general format so regardless of endianess will gives the same result. That is both the endianess will give the same result for bit shifting operation after load into the processor.

character constant = one byte representation
Examples:
1. 'a'    - one byte by a character
2. '\x11' - by hexadecimal
3. '\o10' - by octal

9. No.of bits set for the given number

void main()
{
    static int set_bit[] =
    {0, 1, 1, 2, 1, 2, 2, 3, 1, 2, 2, 3, 2, 3, 3, 4};

    int bitcount(unsigned int u)
    {
        int n = 0;
        for(; u != 0; u >>= 4)
            n += set_bit[u & 0x0f];
        return n;
    }

    int k=bitcount(4096);  // Give any value you like
    printf("No of Bits Set for given the number: 4096 is :%d\n",k);
}
