#include <iostream>
#include <sstream>
#include <string>
#include <stack>
using namespace std;

// Extendable enum: for the number System(B10,B16)
typedef enum base
{
        DECI = 0,
        HEX,
        Invalid,
};

// Extendable string to int conversion function 
int StrToInt(char *token, base b)
{
        stringstream conv;
        int integer;

        if(token != NULL)
        {
                switch(b)
                {
                case DECI:
                        conv << token;
                        break;

                case HEX:
                        conv << std::hex << token;
                        break;

                default:
                        return 0;
                }
                conv >> integer;
                conv.clear();
        }
        return integer;
}

int main ()
{
        stack<int> stk;
        int opCode =0,data =0;
        int a=0,b=0;
        string CliffordCodeStr;

        cout<<"Enter the Clifford Code: ";
        getline (cin,CliffordCodeStr);
        char *ClifCode = const_cast<char*>(CliffordCodeStr.c_str());
        char *token;
        token = strtok(ClifCode," ");

        while(token!=NULL)
        {
          //opCode = StrToInt(token,DECI);
                opCode = StrToInt(token,HEX);
                token = strtok(NULL," ");

                if(token!=NULL)
                  //data = StrToInt(token,DECI);
                    data = StrToInt(token,HEX);
                else
                        data = 0;

                switch(opCode)
                {
                case 0: // push
                        stk.push(data);
                        break;

                case 1: // pop
                        stk.pop();
                        break;

                case 2: // add
                        if((int)stk.size() > 1)
                        {
                                a = stk.top ();
                                stk.pop();
                                b = stk.top();
                                stk.pop();
                        }
                        else if ((int)stk.size() == 1)
                        {
                                a = stk.top ();
                                stk.pop();
                                b = 0;
                        }
                        else
                        {
                                a = b = 0;
                        }
                        stk.push(a+b);
                        a = b =0;
                        break;

                case 3: // sub
                        if((int)stk.size() > 1)
                        {
                                a = stk.top ();
                                stk.pop();
                                b = stk.top();
                                stk.pop();
                        }
                        else if ((int)stk.size() == 1)
                        {
                                a = stk.top ();
                                stk.pop();
                                b = 0;
                        }
                        else
                        {
                                a = b = 0;
                        }
                        stk.push(b-a);
                        a = b =0;
                        break;

                case 4: // mul
                        if((int)stk.size() > 1)
                        {
                                a = stk.top ();
                                stk.pop();
                                b = stk.top();
                                stk.pop();
                        }
                        else if ((int)stk.size() == 1)
                        {
                                a = stk.top ();
                                stk.pop();
                                b = 1;
                        }
                        else
                        {
                                a = b = 0;
                        }
                        stk.push(b*a);
                        a = b =0;
                        break;

                case 5: // div
                        if((int)stk.size() > 1)
                        {
                                a = stk.top ();
                                stk.pop();
                                b = stk.top();
                                stk.pop();
                        }
                        else if ((int)stk.size() == 1)
                        {
                                a = stk.top ();
                                stk.pop();
                                b = a * a ;
                        }
                        else
                        {
                                a = 1;
                                b = 0;
                        }
                        if(a!=0)
                                stk.push(b/a);
                        else
                        {
                                cout<<"cannot divide by zero";
                                return 0;
                        }
                        a = b =0;
                        break;

                default:
                        return 0;
                }
                token = strtok(NULL," ");
        }
        cout<<"Result on the top of the stack(HEX): " <<std::hex<< stk.top();
        return 0;
}
