Non-type template parameter (NTP)




// Ref: https://www.udemy.com/course/beg-cpp-temp/learn/lecture/6958196
// Template Arugment deduction and Instantiation by Umar Lone from Udemy

     Template instantiation will take place only
         1. a function template is invoked
         2. taking address of a function template
         3. using explicit instantiation
         4. creating explicit specialization.

    // Example for 1: a function template is invoked.

    // int Max(int x, int y)       { return x < y ? x : y; }
    // float Max(float x, float y) { return x < y ? x : y; }

    // Function Template:
    template <typename T> // Argument list: Override the deduction by specifiying the explicit type.
    T Max(T x, T y)       // Argument deduction: 'T' from 'T x' decides whole template type
    { 
        cout << typeid(T).name() << endl;
        return x > y ? x : y; 
    }

    // Ref: https://www.udemy.com/course/beg-cpp-temp/learn/lecture/6958198
    // Example for 3: using 'explicit instantiation'
    // Template instatiated by this function declaration even though we did NOT called that function definition
    template char Max(char x, char y);

    // Example for 4: creating 'explicit specialization'.
    // NOTE: This definition SHOULD be in '.cpp' files NOT in '.h' - Bcos it is already instantiatied
    template <> const char* Max<const char*>(const char *x, const char *y)
 // template <> const char* Max(const char *x, const char *y) // same as above
    {
        return strcmp(x, y) > 0 ? x : y;
    }

    int main() {
        // Type name is resolved by 'arugment deduction'
        auto num = Max(3.3f, 5.5f);
        cout << num << endl;
        num = Max(3, 5);
        cout << num << endl;

        auto mixed = Max(3, 5.5f); // ERROR: mixed type 
        cout << mixed;

        auto mixed = Max(static_cast<float> (3), 5.5f); // OK
        cout << mixed;

        // Type name is resolved by 'Argument list: Override deduction by specifying the explicit type'
        auto explicitly = Max<float>(3, 5.5f); // explicitly tell T is float, NOT by argument deduction        
        cout << explicitly;

    /*
        Template instantiation will take place only
            1. a function template is invoked
            2. taking address of a function template
            3. using explicit instantiation
            4. creating explicit specialization.
    */

        // Example for 2: taking address of a function template
        int(*pfn) (int, int) = Max; // Template instatiated even though we did NOT called that function.

        // Example for 4: creating explicit specialization.
        const char *a{ "A" };
        const char *b{ "B" };
        // explicit specialization for string comparison gets called. 
        auto s = Max(a, b);     
        // If don't have that function definition then a regular 
        // template function gets called which gives a wrong result 
        // bcos it compare address values NOT the content.
        cout << s;
    }

    // Ref: https://www.udemy.com/course/beg-cpp-temp/learn/lecture/6958200
    // Non-type template parameter (NTP)
    template<int n>
    void show() {
        /// n = n + 1;  // ERROR: You can NOT change it
        cout << n;
        char buffer[n]; // OK. Advantage: No need for creating a 'dynamic array of size N'
    }

    int main() {
        // NTP (Passing CONST EXP[compile time computed constant value])
        show<9>(); // 9

        // CONST EXP: 'addresses','references','integrals','nullptr','enums'
        int i = 9;
        // show<i>(); // ERROR: Should be constant
        show<sizeof i>(); // 4
    }

    template<typename T>
    T Sum_without_nonTypeTempArg(T *pArr, int sz) { // sz - passed by caller
        T sum{};
        for (int i = 0; i < sz; ++i) sum += pArr[i];
        return sum;
    }

    int main() {
        int n = 0;
        int arr[]{ 1,2,3,4,5 };
        // without using 'NTP'
        n = Sum_without_nonTypeTempArg(arr, 5);
    } 

    template<typename T, int sz> // sz - NO need to pass by the caller. 
                     // It is calculated automatically by 'reference to array'. 
    T Sum_with_nonTypeTempArg(T (&pArr)[sz]) {
        T sum{};
        for (int i = 0; i < sz; ++i) sum += pArr[i];
        return sum;
    }

    int main() {
        int arr[]{ 1,2,3,4,5 };
        // Reference to Array - Implicitly knows the size of an array
        // int(&ref)[5] = arr; // OK
        // int(&ref)[6] = arr; // ERROR - Array size should be the exact size

        // Taking advantage of 'reference to array' and 'NTP'
        auto n = Sum_with_nonTypeTempArg(arr); // array size NOT needed
        cout << n;

        // Note: global std library function - begin(arr) - Use the 
        // advantage of 'reference to array' and 'NTP'
        auto it = std::begin(arr);
