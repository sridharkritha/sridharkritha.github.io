Reference/Pointer



1. References: References are the type of C++ variable that acts as an alias to another variable.
Advantage:
    1. References always “point” to valid objects, and can never be pointed to deallocated memory, references are safer to use than pointers.
    2. We can move the pointer to the next entity by increment but it can be done on reference.
    3. Increase the readability of the program.

2. How to initialize a reference variable? 
    Reference variable should be defined at the place of declaration otherwise it will give a compilation error.
        int nValue   = 5;      // normal integer
        int &rnRef;            // Error, needs to reference something
        int &rnRef   = nValue; // valid reference, rnRef alias of nValue
        int &rnRef   = 5;      // Error, should have const keyword
    const int &rnRef = 5;      // Valid

Calling the function with reference argument:
    void Swap(int& a, int& b) { /* swap operation */ }

    int main() {
        int x = 5, y = 8;
        Swap(x, y);   // Valid
        Swap(5, 8);   // Error - should be variable rather than constant literal
                      // Works for Swap(const int& a, const int& b) but you can NOT swap a & b
    }

3. Is there a situation when I can declare a reference variable without an initializer?
    There are 4 situations for this:

    1. Function declaration
                void foo(int&);

    2. Function return type declaration    
                int& foo();

    3. Reference-type class member
                class foo { int &val; };

     4. Declaration of extern variable
                extern int &val;

4. Use of const in reference variable ?
    Apart from making the reference variable is constant forever the keyword const makes the reference variable can act as an alias of another variable which is a different data type from reference variable type by implicit type conversion.
    int intVar  = 5;
    long &ref1 = intVar;          // Error -  because of incompatible type assignment
    const long &ref2 = intVar;    // OK - Implicit type conversion by const

Reference to const  ( const & ):
    int x;
    const int &ref = x;
    ref = 5; // Error - NOT allows to change the value of x through 'ref' but 'x' CAN be changed directly.

Reference to const - plays an important role in function parameters - which guarantees that the value of the input parameter will not be changed inside the function.
    void foo(const myClass &arg) { }


Notes: Complex combinations
1. & Const  == &
2. Const & = Ref constant
3. * & = Valid - Ref to pointer
4. & * = Invalid - this is NOT pointer to reference

Examples:
    void main() {
        int nValue = 5;      // normal integer
        int &rnRef;          // Error, needs to reference something
        int &rnRef = nValue; // valid reference, rnRef alias of nValue

        nValue = 10;
        cout << rnRef;       // prints 10

        cout << &nValue;     // prints 0012FF7C
        cout << &rnRef;      // prints 0012FF7C

        int temp = 15;
        rnRef = temp;        // assigns value 15 to nValue 
    
        cout << rnRef;       // prints 15
    }

5. References == const pointer :
    void main() {
        int nValue = 5;
        int &ref = nValue;        // OK: reference should be defined at declaration

        // Const. Pointer
        int *const ptr = &nValue; // OK: Const ptr should be defined at declaration

        *ptr = 10;
        cout<<nValue<<endl;       // prints 10
        cout<<ref<<endl;          // prints 10

        int temp = 20;
        ptr = &temp;              // Error,because const ptr !

        // Pointer to Constant value
        const int *pToConst = &nValue;
        pToConst = &temp;            // valid
        *pToConst = 30;              // Error, because points to const value
        temp = 40;                   // valid, we can change the value using other than the pointer variable *pToConst
        cout<<*pToConst<<endl;       // prints 40

        // Reference to Const
        int local = 50;
        const int &rnRef = local;
        rnRef = 60;                  // Error, rnRef is const
        local = 70;                  // valid, we can change the value using other than the ref variable rnRef
        cout<<rnRef<<endl;           // prints 70
    }

Dereferencing the pointer: The new operator returns a pointer to an area of memory that holds an object, it is possible to refer to the original object by dereferencing the pointer.
    int &ref = *(new int)
    ref = 5;

Invalid operations:
    int x = 3;
    int &ref = x;
    int &&dref = ref ;     // Error - ref to ref is invalid - ref is valid only for ONE LEVEL of reference
    int &*refptr = &ref;   // Error - pointer to ref is invalid

Pointer to reference: This will points directly to the variable itself.
    int x = 5;
    int &ref = x;
    int *ptr = &ref;
    *ptr = 100;
 
    Note:
        1. ptr == ref      ==> Wrong assumption bcos types are different (addr and variable value comparision)
        2. ptr == &ref     ==> Valid
        3. ptr == &x       ==> Valid

Reference to pointer: You can create ref to any type including pointers
    int *ptr;
    int *&ref = ptr;
    ref = new int;
    *ref =5;

Reference to Dynamic Objects:
Reference to dynamic object can be created by dereferencing the address returned by the new operator when creating the dynamic object.
    X &ref = * ( new X )

Ex: 
    class student {
        private:
            char m_name[25];
            int m_mark;
        public:
            void setData(int mark, char *name)
            {
                strcpy(m_name, name);
                m_mark = mark;
            }
            void showData() { cout<< m_name << m_mark ; }
    };

    int main() {
        student &ref = * (new student);// ref = reference to a dynamic object
        ref.setData(425, "Rahul");   // using dynamic object like a static object variable (no need for either -> or *)
        ref.showData();      // Rahul  425
        student S;
        S.setData(376, "John");
        student &sRef = S;   // sRef = reference to stack object
        sRef.showData();  // John 376
    }

Memory Allocation and Deallocation using reference:
    struct object {
        float &xPosition;
        float &yPosition;
        float &zPosition;

        object(float &xPos, float &yPos, float &zPos)
            :xPosition(xPos),
             yPosition(yPos),
             zPosition(zPos),
             allocated(false) { }

        object()
            :xPosition(*new float(0.0f)),
             yPosition(*new float(0.0f)),
             zPosition(*new float(0.0f)),
             allocated(true) { }

        ~object() {
            if(allocated) {
                delete &xPosition;
                delete &yPosition;
                delete &zPosition;
            }
        }

    private:
        bool allocated;
    };
Note: Memory allocated to such dynamic object cannot be released except during the termination of the program.

Passing Object as Argument:   1. Pass by Value 
                                                 2. Pass by Reference and 
                                                 3. Pass by Pointer.
1. Pass by Value:
    class sum {
    private:
        int a;
    public:
        sum(int x) { a = x; }
        void add(sum obj1, sum obj2);
    };
    void sum::add(sum obj1, sum obj2) { cout<< (obj1.a + obj2.a); }

    int main() {
        sum obj1(3), obj2(5),obj3;
        obj3.add(obj1, obj2);
    }

2. Pass by Reference:
    void sum::add(sum &obj1, sum &obj2) { cout<< (obj1.a + obj2.a); }

    int main() {
        sum obj1(3), obj2(5),obj3;
        obj3.add(obj1, obj2);
    }

3. Pass by Pointer:
    void sum::add(sum *obj1, sum *obj2) { cout<< (obj1->a + obj2->a); }

    int main() {
        sum obj1(3), obj2(5),obj3;
        obj3.add(&obj1, &obj2);
    }
